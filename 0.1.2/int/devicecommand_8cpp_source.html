108/*!
109 * Checks if \a info is the device (if any) we're looking for, and if so, create a contoller and
110 * service, and begins connecting to the device.
111 */
112void DeviceCommand::deviceDiscovered(const QBluetoothDeviceInfo &info)
113{
114 if (device) {
115 qCDebug(lc).noquote() << tr("Ignoring additional Pokit device \"%1\" (%2) at (%3).")
116 .arg(info.name(), info.deviceUuid().toString(), info.address().toString());
117 return;
118 }
119
120 if ((deviceToScanFor.isEmpty()) || (deviceToScanFor == info.name()) ||
121 ((!info.address().isNull()) && (info.address() == QBluetoothAddress(deviceToScanFor))) ||
122 ((!info.deviceUuid().isNull()) && (info.deviceUuid() == QBluetoothUuid(deviceToScanFor))))
123 {
124 qCDebug(lc).noquote() << tr("Found Pokit device \"%1\" (%2) at (%3).")
125 .arg(info.name(), info.deviceUuid().toString(), info.address().toString());
126 discoveryAgent->stop();
127
128 device = new PokitDevice(info, this);
129 connect(device->controller(), &QLowEnergyController::disconnected,
130 this, &DeviceCommand::deviceDisconnected);
131 connect(device->controller(),
132 #if (QT_VERSION < QT_VERSION_CHECK(6, 2, 0))
133 QOverload<QLowEnergyController::Error>::of(&QLowEnergyController::error),
134 #else
135 &QLowEnergyController::errorOccurred,
136 #endif
137 this, &DeviceCommand::controllerError, Qt::QueuedConnection);
138
139 AbstractPokitService * const service = getService();
140
141 Q_ASSERT(service);
142 connect(service, &AbstractPokitService::serviceDetailsDiscovered,
143 this, &DeviceCommand::serviceDetailsDiscovered);
144 connect(service, &AbstractPokitService::serviceErrorOccurred,
145 this, &DeviceCommand::serviceError);
146
147 qCDebug(lc).noquote() << tr("Connecting to Pokit device \"%1\" (%2) at (%3).")
148 .arg(info.name(), info.deviceUuid().toString(), info.address().toString());
149 device->controller()->connectToDevice();
150 return;
151 }
152
153 qCDebug(lc).noquote() << tr("Ignoring non-matching Pokit device \"%1\" (%2) at (%3).")
154 .arg(info.name(), info.deviceUuid().toString(), info.address().toString());
155 return;
156}
