57AbstractPokitService * FlashLedCommand::getService()
58{
59 Q_ASSERT(device);
60 if (!service) {
61 service = device->status();
62 Q_ASSERT(service);
63 connect(service, &StatusService::deviceLedFlashed,
64 this, &FlashLedCommand::deviceLedFlashed);
65 }
66 return service;
67}

83/*!
84 * Handles StatusService::deviceLedFlashed events, by outputting the result and exiting.
85 */
86void FlashLedCommand::deviceLedFlashed()
87{
88 switch (format) {
89 case OutputFormat::Csv:
90 std::cout << qUtf8Printable(tr("flash_led_result\nsuccess\n"));
91 break;
92 case OutputFormat::Json:
93 std::cout << qUtf8Printable(QLatin1String("true\n"));
94 break;
95 case OutputFormat::Text:
96 std::cout << qUtf8Printable(tr("Done.\n"));
97 break;
98 }
99 if (device) disconnect(); // Will exit the application once disconnected.
100}
