192/*!
193 * Returns the most recent value of the `Status` service's `Device Characteristics` characteristic.
194 *
195 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
196 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), then a
197 * null result is returned, which can be checked via the returned
198 * DeviceCharacteristics::firmwareVersion, like:
199 *
200 * ```
201 * const DeviceCharacteristics characteristics = service->deviceCharacteristics();
202 * if (!characteristics.firmwareVersion.isNull()) {
203 * ...
204 * }
205 * ```
206 */
207StatusService::DeviceCharacteristics StatusService::deviceCharacteristics() const
208{
209 Q_D(const StatusService);
210 const QLowEnergyCharacteristic characteristic =
211 d->getCharacteristic(CharacteristicUuids::deviceCharacteristics);
212 return (characteristic.isValid())
213 ? StatusServicePrivate::parseDeviceCharacteristics(characteristic.value())
214 : StatusService::DeviceCharacteristics();
215}
216
217/*!
218 * Returns the most recent value of the `Status` service's `Status` characteristic.
219 *
220 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
221 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), then the
222 * returned StatusService::Status::batteryLevel member will be a quiet NaN, which can be checked
223 * like:
224 *
225 * ```
226 * const StatusService::Status status = statusService->status();
227 * if (qIsNaN(status.batteryVoltage)) {
228 * // Handle failure.
229 * }
230 * ```
231 */
232StatusService::Status StatusService::status() const
233{
234 Q_D(const StatusService);
235 const QLowEnergyCharacteristic characteristic =
236 d->getCharacteristic(CharacteristicUuids::status);
237 return (characteristic.isValid()) ? StatusServicePrivate::parseStatus(characteristic.value())
238 : StatusService::Status{ DeviceStatus::Idle, std::numeric_limits<float>::quiet_NaN(),
239 BatteryStatus::Low };
240}

257/*!
258 * Set's the Pokit device's name to \a name.
259 *
260 * Returns `true` if the write request was successfully queued, `false` otherwise.
261 *
262 * Emits deviceNameWritten() if/when the \a name has been set.
263 */
264bool StatusService::setDeviceName(const QString &name)
265{
266 Q_D(const StatusService);
267 const QLowEnergyCharacteristic characteristic =
268 d->getCharacteristic(CharacteristicUuids::name);
269 if (!characteristic.isValid()) {
270 return false;
271 }
272
273 const QByteArray value = name.toUtf8();
274 if (value.length() > 11) {
275 qCWarning(d->lc).noquote() << tr("Device name \"%1\" is too long (%2 > 11 bytes): 0x3")
276 .arg(name).arg(value.length()).arg(QLatin1String(value.toHex()));
277 return false;
278 }
279
280 d->service->writeCharacteristic(characteristic, value);
281 return (d->service->error() != QLowEnergyService::ServiceError::CharacteristicWriteError);
282}

284/*!
285 * Flash the Pokit device's LED.
286 *
287 * Returns `true` if the flash request was successfully queued, `false` otherwise.
288 *
289 * Emits deviceLedFlashed() if/when the LED has flashed successfully.
290 *
291 * \note This operation is only supported by Pokit Meter devices. Pokit Pro devices will report an
292 * Bluetooth ATT error `0x80`.
293 *
294 * \cond internal
295 * \pokitApi The Android app can turn Pokit Pro LEDs on/off. Perhaps that is handled by an
296 * undocumented use of this characteristic. Or perhaps its via some other service.
297 * \endcond
298 */
299bool StatusService::flashLed()
300{
301 Q_D(const StatusService);
302 const QLowEnergyCharacteristic characteristic =
303 d->getCharacteristic(CharacteristicUuids::flashLed);
304 if (!characteristic.isValid()) {
305 return false;
306 }
307
308 // The Flash LED characeristic is write-only, and takes a single uint8 "LED" parameter, which
309 // must always be 1. Presumably this is an index for which LED to flash, but the Pokit API docs
310 // say that "any value other than 1 will be ignored", which makes sense given that all current
311 // Pokit devices have only one LED.
312 const QByteArray value(1, '\x01');
313 d->service->writeCharacteristic(characteristic, value);
314 return (d->service->error() != QLowEnergyService::ServiceError::CharacteristicWriteError);
315}

375/*!
376 * Parses the `Device Characteristics` \a value into a DeviceCharacteristics struct.
377 */
378StatusService::DeviceCharacteristics StatusServicePrivate::parseDeviceCharacteristics(
379 const QByteArray &value)
380{
381 StatusService::DeviceCharacteristics characteristics{
382 QVersionNumber(), 0, 0, 0, 0, 0, 0, QBluetoothAddress()
383 };
384 Q_ASSERT(characteristics.firmwareVersion.isNull()); // How we indicate failure.
385
386 if (!checkSize(QLatin1String("Device Characterisitcs"), value, 20, 20)) {
387 return characteristics;
388 }
389
390 characteristics.firmwareVersion = QVersionNumber(
391 qFromLittleEndian<quint8 >(value.mid(0,1)),
392 qFromLittleEndian<quint8 >(value.mid(1,1)));
393 characteristics.maximumVoltage = qFromLittleEndian<quint16>(value.mid(2,2));
394 characteristics.maximumCurrent = qFromLittleEndian<quint16>(value.mid(4,2));
395 characteristics.maximumResistance = qFromLittleEndian<quint16>(value.mid(6,2));
396 characteristics.maximumSamplingRate = qFromLittleEndian<quint16>(value.mid(8,2));
397 characteristics.samplingBufferSize = qFromLittleEndian<quint16>(value.mid(10,2));
398 characteristics.capabilityMask = qFromLittleEndian<quint16>(value.mid(12,2));
399 characteristics.macAddress = QBluetoothAddress(qFromBigEndian<quint64>
400 (QByteArray(2, '\0') + value.mid(14,6)));
401
402 qCDebug(lc).noquote() << tr("Firmware version: ") << characteristics.firmwareVersion;
403 qCDebug(lc).noquote() << tr("Maximum voltage: ") << characteristics.maximumVoltage;
404 qCDebug(lc).noquote() << tr("Maximum current: ") << characteristics.maximumCurrent;
405 qCDebug(lc).noquote() << tr("Maximum resistance: ") << characteristics.maximumResistance;
406 qCDebug(lc).noquote() << tr("Maximum sampling rate:") << characteristics.maximumSamplingRate;
407 qCDebug(lc).noquote() << tr("Sampling buffer size: ") << characteristics.samplingBufferSize;
408 qCDebug(lc).noquote() << tr("Capability mask: ") << characteristics.capabilityMask;
409 qCDebug(lc).noquote() << tr("MAC address: ") << characteristics.macAddress;
410
411 Q_ASSERT(!characteristics.firmwareVersion.isNull()); // How we indicate success.
412 return characteristics;
413}

415/*!
416 * Parses the `Status` \a value into Statu struct. Note, not all Pokit devices support all members
417 * in Status. Specifically, the batteryStatus member is not usually set by Pokit Meter devices, so
418 * will be an invlalid BatteryStatus enum value (`255`) in that case.
419 */
420StatusService::Status StatusServicePrivate::parseStatus(const QByteArray &value)
421{
422 StatusService::Status status{
423 static_cast<StatusService::DeviceStatus>
424 (std::numeric_limits<std::underlying_type<StatusService::DeviceStatus>::type>::max()),
425 std::numeric_limits<float>::quiet_NaN(),
426 static_cast<StatusService::BatteryStatus>
427 (std::numeric_limits<std::underlying_type<StatusService::BatteryStatus>::type>::max()),
428 };
429
430 /*!
431 * \pokitApi Pokit API 0.02 says the `Status` characteristic is 5 bytes. API 1.00 then added an
432 * additional byte for `Battery Status`, for 6 bytes in total. However, Pokit Pro devices return
433 * 8 bytes here. The purpose of those last 2 bytes are not currently known. Note also, Pokit
434 * Meter only uses the first 5 bytes - ie `Battery Status` is not present.
435 */
436
437 if (!checkSize(QLatin1String("Status"), value, 5, 6)) {
438 return status;
439 }
440
441 status.deviceStatus = static_cast<StatusService::DeviceStatus>(value.at(0));
442 status.batteryVoltage = qFromLittleEndian<float>(value.mid(1,4));
443 if (value.size() >= 6) { // Battery Status added to Pokit API docs v1.00.
444 status.batteryStatus = static_cast<StatusService::BatteryStatus>(value.at(5));
445 }
446 qCDebug(lc).noquote() << tr("Device status: %1 (%2)")
447 .arg((quint8)status.deviceStatus).arg(StatusService::toString(status.deviceStatus));
448 qCDebug(lc).noquote() << tr("Battery voltage: %1 volts").arg(status.batteryVoltage);
449 qCDebug(lc).noquote() << tr("Battery status: %1 (%2)")
450 .arg((quint8)status.batteryStatus).arg(StatusService::toString(status.batteryStatus));
451 return status;
452}

454/*!
455 * Handles `QLowEnergyController::serviceDiscovered` events.
456 *
457 * Here we override the base implementation to detect if we're looking at a Pokit Meter, or Pokit
458 * Pro device, as the two devices have very slightly different Status Service UUIDs.
459 */
460void StatusServicePrivate::serviceDiscovered(const QBluetoothUuid &newService)
461{
462 if (newService == StatusService::ServiceUuids::pokitMeter) {
463 qCDebug(lc).noquote() << tr("Found Status Service for a Pokit Meter device.");
464 serviceUuid = StatusService::ServiceUuids::pokitMeter;
465 } else if (newService == StatusService::ServiceUuids::pokitPro) {
466 qCDebug(lc).noquote() << tr("Found Status Service for a Pokit Pro device.");
467 serviceUuid = StatusService::ServiceUuids::pokitPro;
468 }
469 AbstractPokitServicePrivate::serviceDiscovered(newService);
470}

472/*!
473 * Implements AbstractPokitServicePrivate::characteristicRead to parse \a value, then emit a
474 * specialised signal, for each supported \a characteristic.
475 */
476void StatusServicePrivate::characteristicRead(const QLowEnergyCharacteristic &characteristic,
477 const QByteArray &value)
478{
479 AbstractPokitServicePrivate::characteristicRead(characteristic, value);
480
481 Q_Q(StatusService);
482 if (characteristic.uuid() == StatusService::CharacteristicUuids::deviceCharacteristics) {
483 emit q->deviceCharacteristicsRead(parseDeviceCharacteristics(value));
484 return;
485 }
486
487 if (characteristic.uuid() == StatusService::CharacteristicUuids::status) {
488 emit q->deviceStatusRead(parseStatus(value));
489 return;
490 }
491
492 if (characteristic.uuid() == StatusService::CharacteristicUuids::name) {
493 const QString deviceName = QString::fromUtf8(value);
494 qCDebug(lc).noquote() << tr("Device name: \"%1\"").arg(deviceName);
495 emit q->deviceNameRead(deviceName);
496 return;
497 }
498
499 if (characteristic.uuid() == StatusService::CharacteristicUuids::flashLed) {
500 qCWarning(lc).noquote() << tr("Flash LED characteristic is write-only, but somehow read")
501 << serviceUuid << characteristic.name() << characteristic.uuid();
502 return;
503 }
504
505 qCWarning(lc).noquote() << tr("Unknown characteristic read for Status service")
506 << serviceUuid << characteristic.name() << characteristic.uuid();
507}

509/*!
510 * Implements AbstractPokitServicePrivate::characteristicWritten to parse \a newValue, then emit a
511 * specialised signal, for each supported \a characteristic.
512 */
513void StatusServicePrivate::characteristicWritten(const QLowEnergyCharacteristic &characteristic,
514 const QByteArray &newValue)
515{
516 AbstractPokitServicePrivate::characteristicWritten(characteristic, newValue);
517
518 Q_Q(StatusService);
519 if (characteristic.uuid() == StatusService::CharacteristicUuids::deviceCharacteristics) {
520 qCWarning(lc).noquote() << tr("Device Characteristics is read-only, but somehow written")
521 << serviceUuid << characteristic.name() << characteristic.uuid();
522 return;
523 }
524
525 if (characteristic.uuid() == StatusService::CharacteristicUuids::status) {
526 qCWarning(lc).noquote() << tr("Status characteristic is read-only, but somehow written")
527 << serviceUuid << characteristic.name() << characteristic.uuid();
528 return;
529 }
530
531 if (characteristic.uuid() == StatusService::CharacteristicUuids::name) {
532 emit q->deviceNameWritten();
533 return;
534 }
535
536 if (characteristic.uuid() == StatusService::CharacteristicUuids::flashLed) {
537 emit q->deviceLedFlashed();
538 return;
539 }
540
541 qCWarning(lc).noquote() << tr("Unknown characteristic written for Status service")
542 << serviceUuid << characteristic.name() << characteristic.uuid();
543}
