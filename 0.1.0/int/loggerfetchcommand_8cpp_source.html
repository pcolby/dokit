29/*!
30 * \copybrief DeviceCommand::getService
31 *
32 * This override returns a pointer to a DataLoggerService object.
33 */
34AbstractPokitService * LoggerFetchCommand::getService()
35{
36 Q_ASSERT(device);
37 if (!service) {
38 service = device->dataLogger();
39 Q_ASSERT(service);
40 connect(service, &DataLoggerService::metadataRead, this, &LoggerFetchCommand::metadataRead);
41 connect(service, &DataLoggerService::samplesRead, this, &LoggerFetchCommand::outputSamples);
42 }
43 return service;
44}

70 qCDebug(lc) << "updateInterval:" << (int)metadata.updateInterval;
71 qCDebug(lc) << "numberOfSamples:" << metadata.numberOfSamples;
72 qCDebug(lc) << "timestamp:" << metadata.timestamp
73 << QDateTime::fromSecsSinceEpoch(metadata.timestamp, Qt::UTC);
74 this->metadata = metadata;
75 this->samplesToGo = metadata.numberOfSamples;
76 this->timestamp = (quint64)metadata.timestamp * (quint64)1000;
77 qCInfo(lc).noquote() << tr("Fetching %L1 logger samples...").arg(metadata.numberOfSamples);

80/*!
81 * Outputs logger \a samples in the selected ouput format.
82 */
83void LoggerFetchCommand::outputSamples(const DataLoggerService::Samples &samples)
84{
85 QString unit;
86 switch (metadata.mode) {
87 case DataLoggerService::Mode::DcVoltage: unit = QLatin1String("Vdc"); break;
88 case DataLoggerService::Mode::AcVoltage: unit = QLatin1String("Vac"); break;
89 case DataLoggerService::Mode::DcCurrent: unit = QLatin1String("Adc"); break;
90 case DataLoggerService::Mode::AcCurrent: unit = QLatin1String("Aac"); break;
91 default:
92 qCDebug(lc).noquote() << tr("No known unit for mode %1 \"%2\".").arg((int)metadata.mode)
93 .arg(DataLoggerService::toString(metadata.mode));
94 }
95 const QString range = DataLoggerService::toString(metadata.range, metadata.mode);
96
97 for (const qint16 &sample: samples) {
98 const QString timeString = (metadata.timestamp == 0) ? QString::number(timestamp)
99 : QDateTime::fromMSecsSinceEpoch(timestamp, Qt::UTC).toString(Qt::ISODateWithMs);
100 const float value = sample * metadata.scale;
101 switch (format) {
102 case OutputFormat::Csv:
103 for (; showCsvHeader; showCsvHeader = false) {
104 std::cout << qUtf8Printable(tr("timestamp,value,unit,range\n"));
105 }
106 std::cout << qUtf8Printable(QString::fromLatin1("%1,%2,%3,%4\n")
107 .arg(timeString).arg(value).arg(unit, range));
108 break;
109 case OutputFormat::Json:
110 std::cout << QJsonDocument(QJsonObject{
111 { QLatin1String("timestamp"), timeString },
112 { QLatin1String("value"), value },
113 { QLatin1String("unit"), unit },
114 { QLatin1String("range"), range },
115 { QLatin1String("mode"), DataLoggerService::toString(metadata.mode) },
116 }).toJson().toStdString();
117 break;
118 case OutputFormat::Text:
119 std::cout << qUtf8Printable(tr("%1 %2 %3\n").arg(timeString).arg(value).arg(unit));
120 break;
121 }
122 timestamp += metadata.updateInterval;
123 --samplesToGo;
124 }
125 if (samplesToGo <= 0) {
126 qCInfo(lc).noquote() << tr("Finished fetching %L1 samples (with %L2 to remaining).")
127 .arg(metadata.numberOfSamples).arg(samplesToGo);
128 if (device) disconnect(); // Will exit the application once disconnected.
129 }
