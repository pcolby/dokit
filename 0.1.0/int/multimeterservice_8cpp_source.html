381/*!
382 * Configures the Pokit device's multimeter mode.
383 *
384 * Returns `true` if the write request was successfully queued, `false` otherwise.
385 *
386 * Emits settingsWritten() if/when the \a settings have been writtem successfully.
387 */
388bool MultimeterService::setSettings(const Settings &settings)
389{
390 Q_D(const MultimeterService);
391 const QLowEnergyCharacteristic characteristic =
392 d->getCharacteristic(CharacteristicUuids::settings);
393 if (!characteristic.isValid()) {
394 return false;
395 }
396
397 const QByteArray value = d->encodeSettings(settings);
398 if (value.isNull()) {
399 return false;
400 }
401
402 d->service->writeCharacteristic(characteristic, value);
403 return (d->service->error() != QLowEnergyService::ServiceError::CharacteristicWriteError);
404}

406/*!
407 * Returns the most recent value of the `Multimeter` service's `Reading` characteristic.
408 *
409 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
410 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), then the
411 * returned MultimeterService::Reading::value member will be a quiet NaN, which can be checked like:
412 *
413 * ```
414 * const MultimeterService::Reading reading = multimeterService->reading();
415 * if (qIsNaN(reading.value)) {
416 * // Handle failure.
417 * }
418 * ```
419 */
420MultimeterService::Reading MultimeterService::reading() const
421{
422 Q_D(const MultimeterService);
423 const QLowEnergyCharacteristic characteristic =
424 d->getCharacteristic(CharacteristicUuids::reading);
425 return (characteristic.isValid()) ? MultimeterServicePrivate::parseReading(characteristic.value())
426 : Reading{ MeterStatus::Error, std::numeric_limits<float>::quiet_NaN(),
427 Mode::Idle, VoltageRange::AutoRange };
428}

492/*!
493 * Returns \a settings in the format Pokit devices expect.
494 */
495QByteArray MultimeterServicePrivate::encodeSettings(const MultimeterService::Settings &settings)
496{
497 static_assert(sizeof(settings.mode) == 1, "Expected to be 1 byte.");
498 static_assert(sizeof(settings.range) == 1, "Expected to be 1 byte.");
499 static_assert(sizeof(settings.updateInterval) == 4, "Expected to be 4 bytes.");
500
501 QByteArray value;
502 QDataStream stream(&value, QIODevice::WriteOnly);
503 stream.setByteOrder(QDataStream::LittleEndian);
504 stream.setFloatingPointPrecision(QDataStream::SinglePrecision); // 32-bit floats, not 64-bit.
505 stream << (quint8)settings.mode << (quint8)settings.range.voltageRange << settings.updateInterval;
506
507 Q_ASSERT(value.size() == 6);
508 return value;
509}
510
511/*!
512 * Parses the `Reading` \a value into a MultimeterService::Reading struct.
513 */
514MultimeterService::Reading MultimeterServicePrivate::parseReading(const QByteArray &value)
515{
516 MultimeterService::Reading reading{
517 MultimeterService::MeterStatus::Error,
518 std::numeric_limits<float>::quiet_NaN(),
519 MultimeterService::Mode::Idle,
520 MultimeterService::VoltageRange::AutoRange
521 };
522
523 if (!checkSize(QLatin1String("Reading"), value, 7, 7)) {
524 return reading;
525 }
526
527 reading.status = MultimeterService::MeterStatus(value.at(0));
528 reading.value = qFromLittleEndian<float>(value.mid(1,4));
529 reading.mode = static_cast<MultimeterService::Mode>(value.at(5));
530 reading.range.voltageRange = static_cast<MultimeterService::VoltageRange>(value.at(6));
531 return reading;
532}

534/*!
535 * Implements AbstractPokitServicePrivate::characteristicRead to parse \a value, then emit a
536 * specialised signal, for each supported \a characteristic.
537 */
538void MultimeterServicePrivate::characteristicRead(const QLowEnergyCharacteristic &characteristic,
539 const QByteArray &value)
540{
541 AbstractPokitServicePrivate::characteristicRead(characteristic, value);
542
543 Q_Q(MultimeterService);
544 if (characteristic.uuid() == MultimeterService::CharacteristicUuids::reading) {
545 emit q->readingRead(parseReading(value));
546 return;
547 }
548
549 if (characteristic.uuid() == MultimeterService::CharacteristicUuids::settings) {
550 qCWarning(lc).noquote() << tr("Settings characteristic is write-only, but somehow read")
551 << serviceUuid << characteristic.name() << characteristic.uuid();
552 return;
553 }
554
555 qCWarning(lc).noquote() << tr("Unknown characteristic read for Multimeter service")
556 << serviceUuid << characteristic.name() << characteristic.uuid();
557}
