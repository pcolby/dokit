76bool DeviceInfoService::readCharacteristics()
77{
78 const bool r1 = readFirmwareRevisionCharacteristic();
79 const bool r2 = readHardwareRevisionCharacteristic();
80 const bool r3 = readSoftwareRevisionCharacteristic();
81 const bool r4 = readManufacturerCharacteristics();
82 const bool r5 = readModelNumberCharacteristic();
83 return (r1 && r2 && r3 && r4 && r5);
84}

161/*!
162 * Returns the most recent value of the `Device Info` service's `Manufacturer Name` characteristic.

164 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
165 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), then a

299/*!
300 * Implements AbstractPokitServicePrivate::characteristicRead to parse \a value, then emit a
301 * specialised signal, for each supported \a characteristic.
302 */
303void DeviceInfoServicePrivate::characteristicRead(const QLowEnergyCharacteristic &characteristic,
304 const QByteArray &value)
305{
306 AbstractPokitServicePrivate::characteristicRead(characteristic, value);
307
308 Q_Q(DeviceInfoService);
309 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::manufacturerName) {
310 const QString name = QString::fromUtf8(value);
311 qCDebug(lc).noquote() << tr("Manufacturer name: \"%1\"").arg(name);
312 emit q->manufacturerRead(name);
313 return;
314 }
315
316 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::modelNumber) {
317 const QString model = QString::fromUtf8(value);
318 qCDebug(lc).noquote() << tr("Model number: \"%1\"").arg(model);
319 emit q->modelNumberRead(model);
320 return;
321 }
322
323 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::hardwareRevision) {
324 const QString revision = QString::fromUtf8(value);
325 qCDebug(lc).noquote() << tr("Hardware revision: \"%1\"").arg(revision);
326 emit q->hardwareRevisionRead(revision);
327 return;
328 }
329
330 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::firmwareRevision) {
331 const QString revision = QString::fromUtf8(value);
332 qCDebug(lc).noquote() << tr("Firmware revision: \"%1\"").arg(revision);
333 emit q->firmwareRevisionRead(revision);
334 return;
335 }
336
337 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::softwareRevision) {
338 const QString revision = QString::fromUtf8(value);
339 qCDebug(lc).noquote() << tr("Software revision: \"%1\"").arg(revision);
340 emit q->softwareRevisionRead(revision);
341 return;
342 }
343
344 qCWarning(lc).noquote() << tr("Unknown characteristic read for Device Info service")
345 << serviceUuid << characteristic.name() << characteristic.uuid();
346}
