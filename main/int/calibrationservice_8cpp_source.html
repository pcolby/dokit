73/*!
74 * Set's the Pokit device's name to \a name.
75 *
76 * Returns `true` if the write request was successfully queued, `false` otherwise.
77 *
78 * Emits deviceNameWritten() if/when the \a name has been set.
79 */
80bool CalibrationService::calibrateTemperature(const float ambientTemperature)
81{
82 static_assert(sizeof(float) == 4, "Pokit devices expect 32-bit floats");
83 Q_D(const CalibrationService);
84 const QLowEnergyCharacteristic characteristic =
85 d->getCharacteristic(CharacteristicUuids::temperature);
86 if (!characteristic.isValid()) {
87 return false;
88 }
89
90 const QByteArray newValue = CalibrationServicePrivate::encodeTemperature(ambientTemperature);
91 qCDebug(d->lc).noquote() << tr("Writing new temperature %1 (0x%2).")
92 .arg(ambientTemperature).arg(QLatin1String(newValue.toHex()));
93 d->service->writeCharacteristic(characteristic, newValue);
94 return (d->service->error() != QLowEnergyService::ServiceError::CharacteristicWriteError);
95}

123/*!
124 * Returns \a value in a format Pokit devices expect. Specifically, this just enocdes \a value as
125 * a 32-bit float in litte-endian byte order.
126 */
127QByteArray CalibrationServicePrivate::encodeTemperature(const float value)
128{
129 static_assert(sizeof(value) == 4, "Pokit devices expect 32-bit floats");
130 QByteArray bytes(sizeof(float), '\0');
131 qToLittleEndian<float>(value, bytes.data());
132 return bytes;
133}

135/*!
136 * Implements AbstractPokitServicePrivate::characteristicWritten to parse \a newValue, then emit a
137 * specialised signal, for each supported \a characteristic.
138 */
139void CalibrationServicePrivate::characteristicWritten(const QLowEnergyCharacteristic &characteristic,
140 const QByteArray &newValue)
141{
142 AbstractPokitServicePrivate::characteristicWritten(characteristic, newValue);
143
144 Q_Q(CalibrationService);
145 if (characteristic.uuid() == CalibrationService::CharacteristicUuids::temperature) {
146 emit q->temperatureCalibrated();
147 return;
148 }
149
150 qCWarning(lc).noquote() << tr("Unknown characteristic written for Calibration service")
151 << serviceUuid << characteristic.name() << characteristic.uuid();
152}
