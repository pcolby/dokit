32/*!
33 * Construct a new scanner object with \a parent.
34 */
35Scanner::Scanner(QObject * const parent) : QObject(parent)
36{
37 discoveryAgent = new PokitDeviceDiscoveryAgent(this);
38
39 connect(discoveryAgent, &PokitDeviceDiscoveryAgent::pokitDeviceDiscovered,
40 this, &Scanner::deviceDiscovered);
41
42 connect(discoveryAgent, &PokitDeviceDiscoveryAgent::finished, this, []() {
43 qCDebug(pokitScanner) << "Finished scanning for Pokit devices.";
44 QCoreApplication::quit();
45 });
46}

48/*!
49 * Begins scanning for Pokit devices, with \a timeout, in milliseconds.
50 */
51void Scanner::start(const int timeout)
52{
53 Q_ASSERT(discoveryAgent);
54 qCDebug(pokitScanner).noquote().nospace() << "Scanning for Pokit devices (with "
55 << (timeout ? QLocale().toString(timeout)+QLatin1String("ms") : QLatin1String("no"))
56 << " timeout).";
57 discoveryAgent->setLowEnergyDiscoveryTimeout(timeout);
58 discoveryAgent->start();
59}
