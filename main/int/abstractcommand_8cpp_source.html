91/*!
92 * Returns an RFC 4180 compliant version of \a field. That is, if \a field contains any of the
93 * the below four characters, than any double quotes are escaped (by addition double-quotes), and
94 * the string itself surrounded in double-quotes. Otherwise, \a field is returned verbatim.
95 *
96 * Some examples:
97 * ```
98 * QCOMPARE(escapeCsvField("abc"), "abc"); // Returned unchanged.
99 * QCOMPARE(escapeCsvField("a,c"), "\"a,c\""); // Wrapped in double-quotes.
100 * QCOMPARE(escapeCsvField("a\"c"), "\"a""c\""); // Existing double-quotes doubles, then wrapped.
101 * ```
102 */
103QString AbstractCommand::escapeCsvField(const QString &field)
104{
105 if (field.contains(QLatin1Char(','))||field.contains(QLatin1Char('\r'))||
106 field.contains(QLatin1Char('"'))||field.contains(QLatin1Char('\n')))
107 {
108 return QString::fromLatin1("\"%1\"").arg(
109 QString(field).replace(QLatin1Char('"'), QLatin1String("\"\"")));
110 } else return field;
111}

113/*!
114 * Returns \a value as a number of micros, such as microseconds, or microvolts. The string \a value
115 * may end with the optional \a unit, such as `V` or `s`, which may also be preceded with a SI unit
116 * prefix such as `m` for `milli`. If \a value contains no SI unit prefix, then the result will be
117 * multiplied by 1,000 enough times to be greater than \a sensibleMinimum. This allows for
118 * convenient use like:
119 *
120 * ```
121 * const quin32t timeout = parseMicroValue(parser.value("window"), 's', 500*1000);
122 * ```
123 *
124 * So that an unqalified period like "300" will be assumed to be 300 milliseconds, and not 300
125 * microseconds, while a period like "1000" will be assume to be 1 second.
126 *
127 * If conversion fails for any reason, 0 is returned.
128 */
129quint32 AbstractCommand::parseMicroValue(const QString &value, const QString &unit,
130 const quint32 sensibleMinimum)
131{
132 // Remove the optional (whole) unit suffix.
133 quint32 scale = 0;
134 QString number = value.trimmed();
135 if ((!unit.isEmpty()) && (number.endsWith(unit, Qt::CaseInsensitive))) {
136 number.chop(unit.length());
137 scale = 1000 * 1000;
138 }
139
140 // Parse, and remove, the optional SI unit prefix.
141 if (number.endsWith(QLatin1String("m"))) {
142 number.chop(1);
143 scale = 1000;
144 }
145
146 // Parse, and remove, the optional SI unit prefix.
147 if (number.endsWith(QLatin1String("u"))) {
148 number.chop(1);
149 scale = 1;
150 }
151
152 // Parse the number as an (unsigned) integer.
153 QLocale locale; bool ok;
154 const quint32 integer = locale.toUInt(number, &ok);
155 if (ok) {
156 if ((scale == 0) && (integer != 0)) {
157 for (scale = 1; (integer * scale) < sensibleMinimum; scale *= 1000);
158 }
159 return integer * scale;
160 }
161
162 // Parse the number as a (double) floating point number, and check that it is positive.
163 const double dbl = locale.toDouble(number, &ok);
164 if ((ok) && (dbl > 0)) {
165 if ((scale == 0) && (dbl > 0.0)) {
166 for (scale = 1; (dbl * scale) < sensibleMinimum; scale *= 1000);
167 }
168 return dbl * scale;
169 }
170
171 return 0; // Failed to parse as either integer, or float.
172}

229/*!
230 * Returns \a value as a number, with optional SI unit prefix, and optional \a unit suffix. For
231 * example:
232 *
233 * ```
234 * QCOMPARE(parseWholeValue("1.2Mohm", "ohm"), 1200000);
235 * ```
236 *
237 * If conversion fails for any reason, 0 is returned.
238 */
239quint32 AbstractCommand::parseWholeValue(const QString &value, const QString &unit)
240{
241 // Remove the optional unit suffix.
242 QString number = value.trimmed();
243 if (number.endsWith(unit, Qt::CaseInsensitive)) {
244 number.chop(unit.length());
245 }
246
247 // Parse, and remove, the optional SI unit prefix.
248 quint32 scale = 1;
249 if (number.endsWith(QLatin1String("k"), Qt::CaseInsensitive)) {
250 number.chop(1);
251 scale = 1000;
252 } else if (number.endsWith(QLatin1String("M"))) {
253 number.chop(1);
254 scale = 1000 * 1000;
255 }
256
257 // Parse the number as an (unsigned) integer.
258 QLocale locale; bool ok;
259 const quint16 integer = locale.toUInt(number, &ok);
260 if (ok) {
261 return integer * scale;
262 }
263
264 // Parse the number as a (double) floating point number, and check that it is positive.
265 const double dbl = locale.toDouble(number, &ok);
266 if ((ok) && (dbl > 0)) {
267 return dbl * scale;
268 }
269
270 return 0; // Failed to parse as either integer, or float.
271}

273/*!
274 * Processes the relevant options from the command line \a parser.
275 *
276 * On success, returns an empty QStringList, otherwise returns a list of CLI errors that the caller
277 * should report appropriately before exiting.
278 *
279 * This base implementations performs some common checks, such as ensuring that required options are
280 * present. Derived classes should override this function to perform further processing, typically
281 * inovking this base implementation as a first step, such as:
282 *
283 * ```
284 * QStringList CustomCommand::processOptions(const QCommandLineParser &parser)
285 * {
286 * QStringList errors = AbstractCommand::processOptions(parser);
287 * if (!errors.isEmpty()) {
288 * return errors;
289 * }
290 *
291 * // Do further procession of options.
292 *
293 * return errors;
294 * }
295 * ```
296 */
297QStringList AbstractCommand::processOptions(const QCommandLineParser &parser)
298{
299 // Report any supplied options that are not supported by this command.
300 const QStringList suppliedOptions = parser.optionNames();
301 const QStringList supportedOptions = this->supportedOptions(parser);
302 for (const QString &option: suppliedOptions) {
303 if (!supportedOptions.contains(option)) {
304 qCInfo(lc).noquote() << tr("Ignoring option: %1").arg(option);
305 }
306 }
307 QStringList errors;
308
309 // Parse the device (name/addr/uuid) option.
310 if (parser.isSet(QLatin1String("device"))) {
311 deviceToScanFor = parser.value(QLatin1String("device"));
312 }
313
314 // Parse the output format options (if supported, and supplied).
315 if ((supportedOptions.contains(QLatin1String("output"))) && // Derived classes may have removed.
316 (parser.isSet(QLatin1String("output"))))
317 {
318 const QString output = parser.value(QLatin1String("output")).toLower();
319 if (output == QLatin1String("csv")) {
320 format = OutputFormat::Csv;
321 } else if (output == QLatin1String("json")) {
322 format = OutputFormat::Json;
323 } else if (output == QLatin1String("text")) {
324 format = OutputFormat::Text;
325 } else {
326 errors.append(tr("Unknown output format: %1").arg(output));
327 }
328 }
329
330 // Parse the device scan timeout option.
331 if (parser.isSet(QLatin1String("timeout"))) {
332 const quint32 timeout = parseMilliValue(parser.value(QLatin1String("timeout")),
333 QLatin1String("s"), 500);
334 if (timeout == 0) {
335 errors.append(tr("Invalid timeout: %1").arg(parser.value(QLatin1String("timeout"))));
336 } else {
337 discoveryAgent->setLowEnergyDiscoveryTimeout(timeout);
338 qCDebug(lc).noquote() << tr("Set scan timeout to %1").arg(
339 discoveryAgent->lowEnergyDiscoveryTimeout());
340 }
341 }
342
343 // Return errors for any required options that are absent.
344 const QStringList requiredOptions = this->requiredOptions(parser);
345 for (const QString &option: requiredOptions) {
346 if (!parser.isSet(option)) {
347 errors.append(tr("Missing required option: %1").arg(option));
348 }
349 }
350 return errors;
351}
