81/*!
82 * Returns an RFC 4180 compliant version of \a field. That is, if \a field contains any of the
83 * the below four characters, than any double quotes are escaped (by addition double-quotes), and
84 * the string itself surrounded in double-quotes. Otherwise, \a field is returned verbatim.
85 *
86 * Some examples:
87 * ```
88 * QCOMPARE(escapeCsvField("abc"), "abc"); // Returned unchanged.
89 * QCOMPARE(escapeCsvField("a,c"), "\"a,c\""); // Wrapped in double-quotes.
90 * QCOMPARE(escapeCsvField("a\"c"), "\"a""c\""); // Existing double-quotes doubles, then wrapped.
91 * ```
92 */
93QString AbstractCommand::escapeCsvField(const QString &field)
94{
95 if (field.contains(QLatin1Char(','))||field.contains(QLatin1Char('\r'))||
96 field.contains(QLatin1Char('"'))||field.contains(QLatin1Char('\n')))
97 {
98 return QString::fromLatin1("\"%1\"").arg(
99 QString(field).replace(QLatin1Char('"'), QLatin1String("\"\"")));
100 } else return field;
101}

104/*!
105 * Processes the relevant options from the command line \a parser.
106 *
107 * On success, returns an empty QStringList, otherwise returns a list of CLI errors that the caller
108 * should report appropriately before exiting.
109 *
110 * This base implementations performs some common checks, such as ensuring that required options are
111 * present. Derived classes should override this function to perform further processing, typically
112 * inovking this base implementation as a first step, such as:
113 *
114 * ```
115 * QStringList CustomCommand::processOptions(const QCommandLineParser &parser)
116 * {
117 * QStringList errors = AbstractCommand::processOptions(parser);
118 * if (!errors.isEmpty()) {
119 * return errors;
120 * }
121 *
122 * // Do further procession of options.
123 *
124 * return errors;
125 * }
126 * ```
127 */
128QStringList AbstractCommand::processOptions(const QCommandLineParser &parser)
129{
130 // Report any supplied options that are not supported by this command.
131 const QStringList suppliedOptions = parser.optionNames();
132 const QStringList supportedOptions = this->supportedOptions();
133 for (const QString &option: suppliedOptions) {
134 if (!supportedOptions.contains(option)) {
135 qCInfo(lc).noquote() << tr("Ignoring option: %1").arg(option);
136 }
137 }
138 QStringList errors;
139
140 // Parse the output format options (if supported, and supplied).
141 if ((supportedOptions.contains(QLatin1String("output"))) && // Derived classes may have removed.
142 (parser.isSet(QLatin1String("output"))))
143 {
144 const QString output = parser.value(QLatin1String("output")).toLower();
145 if (output == QLatin1String("csv")) {
146 format = OutputFormat::Csv;
147 } else if (output == QLatin1String("json")) {
148 format = OutputFormat::Json;
149 } else if (output == QLatin1String("text")) {
150 format = OutputFormat::Text;
151 } else {
152 errors.append(tr("Unknown output format: %1").arg(output));
153 }
154 }
155
156 // Return errors for any required options that are absent.
157 const QStringList requiredOptions = this->requiredOptions();
158 for (const QString &option: requiredOptions) {
159 if (!parser.isSet(option)) {
160 errors.append(tr("Missing required option: %1").arg(option));
161 }
162 }
163 return errors;
164}
