153/*!
154 * Returns the most recent value of the `Status` service's `Device Characteristics` characteristic.
155 *
156 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
157 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), then a
158 * null result is returned, which can be checked via the returned
159 * DeviceCharacteristics::firmwareVersion, like:
160 *
161 * ```
162 * const DeviceCharacteristics characteristics = service->deviceCharacteristics();
163 * if (!characteristics.firmwareVersion.isNull()) {
164 * ...
165 * }
166 * ```
167 */
168StatusService::DeviceCharacteristics StatusService::deviceCharacteristics() const
169{
170 Q_D(const StatusService);
171 const QLowEnergyCharacteristic characteristic =
172 d->getCharacteristic(CharacteristicUuids::deviceCharacteristics);
173 return (characteristic.isValid())
174 ? StatusServicePrivate::parseDeviceCharacteristics(characteristic.value())
175 : StatusService::DeviceCharacteristics();
176}

178/*!
179 * Returns the most recent value of the `Status` service's `Status` characteristic.
180 *
181 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
182 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), then the
183 * returned StatusService::Status::batteryLevel member will be a quiet NaN, which can be checked
184 * like:
185 *
186 * ```
187 * const StatusService::Status status = statusService->status();
188 * if (qIsNaN(status.batteryLevel)) {
189 * // Handle failure.
190 * }
191 * ```
192 */
193StatusService::Status StatusService::status() const
194{
195 Q_D(const StatusService);
196 const QLowEnergyCharacteristic characteristic =
197 d->getCharacteristic(CharacteristicUuids::status);
198 return (characteristic.isValid()) ? StatusServicePrivate::parseStatus(characteristic.value())
199 : StatusService::Status{ DeviceStatus::Idle, std::numeric_limits<float>::quiet_NaN(),
200 BatteryStatus::Low };
201}

251/*!
252 * Set's the Pokit device's name to \a name.
253 *
254 * Returns `true` if the write request was successfully queued, `false` otherwise.
255 *
256 * Emits deivceNameWritten() if/when the \a name has been set.
257 */
258bool StatusService::setDeviceName(const QString &name)
259{
260 Q_D(const StatusService);
261 const QLowEnergyCharacteristic characteristic =
262 d->getCharacteristic(CharacteristicUuids::name);
263 if (!characteristic.isValid()) {
264 return false;
265 }
266
267 const QByteArray value = name.toUtf8();
268 if (value.length() > 11) {
269 qCWarning(d->lc).noquote() << tr("Device name \"%1\" is too long (%2 > 11 bytes): 0x3")
270 .arg(name).arg(value.length()).arg(QLatin1String(value.toHex()));
271 return false;
272 }
273
274 d->service->writeCharacteristic(characteristic, value);
275 return (d->service->error() != QLowEnergyService::ServiceError::CharacteristicWriteError);
276}

278/*!
279 * Flash the Pokit device's LED.
280 *
281 * Returns `true` if the flash request was successfully queued, `false` otherwise.
282 *
283 * Emits deviceLedFlashed() if/when the LED has flashed successfully.
284 *
285 * \note This does not appear to work currently, for the one (Pokit Pro) device available for
286 * testing. Instead, the underlying Bluetooth stack returns ATT error `0x80`.
287 *
288 * \todo Diagnose and/or follow up with Pokit Innovations.
289 */
290bool StatusService::flashLed()
291{
292 Q_D(const StatusService);
293 const QLowEnergyCharacteristic characteristic =
294 d->getCharacteristic(CharacteristicUuids::flashLed);
295 if (!characteristic.isValid()) {
296 return false;
297 }
298
299 // The Flash LED characeristic is write-only, and takes a single uint8 "LED" parameter, which
300 // must always be 1. Presumably this is an index for which LED to flash, but the Pokit API docs
301 // say that "any value other than 1 will be ignored", which makes sense given that all current
302 // Pokit devices have only one LED.
303 const QByteArray value(1, '\x01');
304 d->service->writeCharacteristic(characteristic, value);
305 return (d->service->error() != QLowEnergyService::ServiceError::CharacteristicWriteError);
306}

366/*!
367 * Parses the `Device Characteristics` \a value into a DeviceCharacteristics struct.
368 */
369StatusService::DeviceCharacteristics StatusServicePrivate::parseDeviceCharacteristics(
370 const QByteArray &value)
371{
372 StatusService::DeviceCharacteristics characteristics;
373 Q_ASSERT(characteristics.firmwareVersion.isNull()); // How we indicate failure.
374
375 if (value.size() < 20) {
376 qCWarning(lc).noquote() << tr("Invalid characteristics size %1 for value: 0x%2")
377 .arg(value.size()).arg(QLatin1String(value.toHex()));
378 return characteristics;
379 }
380 if (value.size() > 20) {
381 qCWarning(lc).noquote() << tr("Characterisitcs has %1 extra bytes: 0x%2")
382 .arg(value.size()-20).arg(QLatin1String(value.mid(20).toHex()));
383 }
384
385 characteristics.firmwareVersion = QVersionNumber(
386 qFromLittleEndian<quint8 >(value.mid(0,1)),
387 qFromLittleEndian<quint8 >(value.mid(1,1)));
388 characteristics.maximumVoltage = qFromLittleEndian<quint16>(value.mid(2,2));
389 characteristics.maximumCurrent = qFromLittleEndian<quint16>(value.mid(4,2));
390 characteristics.maximumResistance = qFromLittleEndian<quint16>(value.mid(6,2));
391 characteristics.maximumSamplingRate = qFromLittleEndian<quint16>(value.mid(8,2));
392 characteristics.samplingBufferSize = qFromLittleEndian<quint16>(value.mid(10,2));
393 characteristics.capabilityMask = qFromLittleEndian<quint16>(value.mid(12,2));
394 characteristics.macAddress = QBluetoothAddress(qFromBigEndian<quint64>
395 (QByteArray(2, '\0') + value.mid(14,6)));
396
397 qCDebug(lc).noquote() << tr("Firmware version: ") << characteristics.firmwareVersion;
398 qCDebug(lc).noquote() << tr("Maximum voltage: ") << characteristics.maximumVoltage;
399 qCDebug(lc).noquote() << tr("Maximum current: ") << characteristics.maximumCurrent;
400 qCDebug(lc).noquote() << tr("Maximum resistance: ") << characteristics.maximumResistance;
401 qCDebug(lc).noquote() << tr("Maximum sampling rate:") << characteristics.maximumSamplingRate;
402 qCDebug(lc).noquote() << tr("Sampling buffer size: ") << characteristics.samplingBufferSize;
403 qCDebug(lc).noquote() << tr("Capability mask: ") << characteristics.capabilityMask;
404 qCDebug(lc).noquote() << tr("MAC address: ") << characteristics.macAddress;
405
406 Q_ASSERT(!characteristics.firmwareVersion.isNull()); // How we indicate success.
407 return characteristics;
408}

410/*!
411 * Parses the `Status` \a value into a DeviceStatus and battery voltage.
412 */
413StatusService::Status StatusServicePrivate::parseStatus(const QByteArray &value)
414{
415 StatusService::Status status{
416 static_cast<StatusService::DeviceStatus>
417 (std::numeric_limits<StatusService::DeviceStatus>::max()),
418 std::numeric_limits<float>::quiet_NaN(),
419 static_cast<StatusService::BatteryStatus>
420 (std::numeric_limits<StatusService::BatteryStatus>::max()),
421 };
422
423 if (value.size() < 5) {
424 qCWarning(lc).noquote() << tr("Invalid status size %1 for value: 0x%2")
425 .arg(value.size()).arg(QLatin1String(value.toHex()));
426 return status;
427 }
428 if (value.size() > 6) {
429 qCWarning(lc).noquote() << tr("Status has %1 extra bytes: 0x%2")
430 .arg(value.size()-6).arg(QLatin1String(value.mid(6).toHex()));
431 }
432
433 status.deviceStatus = static_cast<StatusService::DeviceStatus>(value.at(0));
434 status.batteryVoltage = qFromLittleEndian<float>(value.mid(1,4));
435 if (value.size() >= 6) { // Battery Status added to Pokit API docs v1.00.
436 status.batteryStatus = static_cast<StatusService::BatteryStatus>(value.at(5));
437 }
438 qCDebug(lc).noquote() << tr("Device status: %1 (%2)")
439 .arg((quint8)status.deviceStatus).arg(StatusService::deviceStatusLabel(status.deviceStatus));
440 qCDebug(lc).noquote() << tr("Battery voltage: %1 volts").arg(status.batteryVoltage);
441 qCDebug(lc).noquote() << tr("Battery status: %1 (%2)")
442 .arg((quint8)status.batteryStatus);
443 return status;
444}

446/*!
447 * Implements AbstractPokitServicePrivate::characteristicRead to parse \a value, then emit a
448 * specialised signal, for each supported \a characteristic.
449 */
450void StatusServicePrivate::characteristicRead(const QLowEnergyCharacteristic &characteristic,
451 const QByteArray &value)
452{
453 qCDebug(lc).noquote() << tr("Read characteristic \"%1\" (%2) of size %3: 0x%4")
454 .arg(characteristic.name(), characteristic.uuid().toString()).arg(value.size())
455 .arg(QLatin1String(value.toHex()));
456
457 Q_Q(StatusService);
458 if (characteristic.uuid() == StatusService::CharacteristicUuids::deviceCharacteristics) {
459 emit q->deviceCharacteristicsRead(parseDeviceCharacteristics(value));
460 return;
461 }
462
463 if (characteristic.uuid() == StatusService::CharacteristicUuids::status) {
464 emit q->deviceStatusRead(parseStatus(value));
465 return;
466 }
467
468 if (characteristic.uuid() == StatusService::CharacteristicUuids::name) {
469 const QString deviceName = QString::fromUtf8(value);
470 qCDebug(lc).noquote() << tr("Device name: \"%1\"").arg(deviceName);
471 emit q->deviceNameRead(deviceName);
472 return;
473 }
474
475 if (characteristic.uuid() == StatusService::CharacteristicUuids::flashLed) {
476 qCWarning(lc).noquote() << tr("Flash LED characteristic is write-only, but somehow read")
477 << serviceUuid << characteristic.name() << characteristic.uuid();
478 return;
479 }
480
481 qCWarning(lc).noquote() << tr("Unknown characteristic read for Status Service")
482 << serviceUuid << characteristic.name() << characteristic.uuid();
483}

485/*!
486 * Implements AbstractPokitServicePrivate::characteristicWritten to parse \a newValue, then emit a
487 * specialised signal, for each supported \a characteristic.
488 */
489void StatusServicePrivate::characteristicWritten(const QLowEnergyCharacteristic &characteristic,
490 const QByteArray &newValue)
491{
492 qCDebug(lc).noquote() << tr("Characteristic \"%1\" (%2) written, with new value:")
493 .arg(characteristic.name(), characteristic.uuid().toString()) << newValue;
494
495 Q_Q(StatusService);
496 if (characteristic.uuid() == StatusService::CharacteristicUuids::deviceCharacteristics) {
497 qCWarning(lc).noquote() << tr("Device Characteristics is read-only, but somehow written")
498 << serviceUuid << characteristic.name() << characteristic.uuid();
499 return;
500 }
501
502 if (characteristic.uuid() == StatusService::CharacteristicUuids::status) {
503 qCWarning(lc).noquote() << tr("Status characteristic is read-only, but somehow written")
504 << serviceUuid << characteristic.name() << characteristic.uuid();
505 return;
506 }
507
508 if (characteristic.uuid() == StatusService::CharacteristicUuids::name) {
509 emit q->deivceNameWritten();
510 return;
511 }
512
513 if (characteristic.uuid() == StatusService::CharacteristicUuids::flashLed) {
514 emit q->deviceLedFlashed();
515 return;
516 }
517
518 qCWarning(lc).noquote() << tr("Unknown characteristic written for Status Service")
519 << serviceUuid << characteristic.name() << characteristic.uuid();
520}
