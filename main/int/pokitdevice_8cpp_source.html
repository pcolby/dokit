37/*!
38 * \class PokitDevice
39 *
40 * The PokitDevice class simplifies Pokit device access.
41 *
42 * It does this by wrapping QLowEnergyController to provide:
43 * * convenient Pokit service factory methods (dataLogger(), deviceInformation(), dso(),
44 * genericAccess(), multimeter() and status()); and
45 * * consistent debug logging of QLowEnergyController events.
46 *
47 * But this class is entirely optional, in that all features of all other QtPokit classes can be
48 * used wihtout this class. It's just a (meaningful) convenience.
49 */
50
51/*!
52 * Constructs a new Pokit device controller wrapper for \a deviceInfo, with \a parent.
53 *
54 * Though not strictly necessary, \a deviceInfo should normally come from a
55 * PokitDiscoveryAgent instance (or a QBluetoothDeviceDiscoveryAgent), otherwise connection
56 * is likely to fail with QLowEnergyController::UnknownRemoteDeviceError.
57 */
58PokitDevice::PokitDevice(const QBluetoothDeviceInfo &deviceInfo, QObject *parent)
59 : QObject(parent), d_ptr(new PokitDevicePrivate(this))
60{
61 Q_D(PokitDevice);
62 d->setController(QLowEnergyController::createCentral(deviceInfo, this));
63}

243/*!
244 * Sets \a newController to be used for accessing Pokit devices.
245 *
246 * If a controller has already been set (and is not the same pointer), then the previous controller
247 * will be disconnected, and replaced with \a newController.
248 *
249 * This function will not take ownership of the new controller. The caller is responsible for
250 * ensuring that \a newContorller remains valid for the lifetime of this instance, or until this
251 * function is used again to replace \a newController with another one (which may be a nullptr).
252 *
253 * \see controller
254 * \see PokitDevice::controller()
255 */
256void PokitDevicePrivate::setController(QLowEnergyController * newController)
257{
258 if (newController == this->controller) {
259 qCDebug(lc).noquote() << tr("Controller already set to:") << newController;
260 return;
261 }
262
263 if (this->controller) {
264 qCDebug(lc).noquote() << tr("Disconnecting signals from previous controller:")
265 << controller;
266 disconnect(this->controller, nullptr, this, nullptr);
267 }
268
269 qCDebug(lc).noquote() << tr("Setting new controller:") << newController;
270 this->controller = newController;
271 if (!newController) {
272 return; // Don't bother continuing to connect if new controller is null.
273 }
274
275 qCDebug(lc).noquote() << tr("Set new controller \"%1\" (%2) at (%3).").arg(
276 controller->remoteName(), controller->remoteDeviceUuid().toString(),
277 controller->remoteAddress().toString());
278
279 connect(controller, &QLowEnergyController::connected,
280 this, &PokitDevicePrivate::connected);
281
282 connect(controller, &QLowEnergyController::connectionUpdated,
283 this, &PokitDevicePrivate::connectionUpdated);
284
285 connect(controller, &QLowEnergyController::disconnected,
286 this, &PokitDevicePrivate::disconnected);
287
288 connect(controller, &QLowEnergyController::discoveryFinished,
289 this, &PokitDevicePrivate::discoveryFinished);
290
291
292 connect(controller,
293 #if (QT_VERSION < QT_VERSION_CHECK(6, 2, 0))
294 QOverload<QLowEnergyController::Error>::of(&QLowEnergyController::error),
295 #else
296 &QLowEnergyController::errorOccurred,
297 #endif
298 this, &PokitDevicePrivate::errorOccurred);
299
300
301 connect(controller, &QLowEnergyController::serviceDiscovered,
302 this, &PokitDevicePrivate::serviceDiscovered);
303
304 connect(controller, &QLowEnergyController::stateChanged,
305 this, &PokitDevicePrivate::stateChanged);
306}
