69/*!
70 * Begins scanning for Pokit devices.
71 */
72bool InfoCommand::start()
73{
74 Q_ASSERT(device);
75 if (!service) {
76 service = device->deviceInformation();
77 Q_ASSERT(service);
78 connect(service, &DeviceInfoService::serviceDetailsDiscovered,
79 this, &InfoCommand::serviceDetailsDiscovered);
80 connect(service, &DeviceInfoService::serviceErrorOccurred,
81 this, &InfoCommand::serviceError);
82 }
83 qCInfo(lc).noquote() << tr("Connecting to device...");
84 device->controller()->connectToDevice();
85 return true;
86}

88/*!
89 * \copybrief DeviceCommand::serviceDetailsDiscovered
90 *
91 * This override fetches the current device's information, and outputs it in the selected format.
92 */
93void InfoCommand::serviceDetailsDiscovered()
94{
95 DeviceCommand::serviceDetailsDiscovered(); // Just logs consistently.
96 switch (format) {
97 case OutputFormat::Csv:
98 fputs(qPrintable(tr("manufacturer_name,model_number,hardware_revision,firmware_revision,"
99 "software_revision\n")), stdout);
100 fputs(qPrintable(QString::fromLatin1("%1,%2,%3,%4,%5\n").arg(
101 escapeCsvField(service->manufacturer()), escapeCsvField(service->modelNumber()),
102 escapeCsvField(service->hardwareRevision()), escapeCsvField(service->firmwareRevision()),
103 escapeCsvField(service->softwareRevision()))), stdout);
104 break;
105 case OutputFormat::Json:
106 fputs(QJsonDocument(QJsonObject{
107 { QLatin1String("manufacturerName"), service->manufacturer() },
108 { QLatin1String("modelNumber"), service->modelNumber() },
109 { QLatin1String("hardwareRevision"), service->hardwareRevision() },
110 { QLatin1String("firmwareRevision"), service->firmwareRevision() },
111 { QLatin1String("softwareRevision"), service->softwareRevision() },
112 }).toJson(), stdout);
113 break;
114 case OutputFormat::Text:
115 fputs(qPrintable(tr("Manufacturer name: %1\n").arg(service->manufacturer())), stdout);
116 fputs(qPrintable(tr("Model number: %1\n").arg(service->modelNumber())), stdout);
117 fputs(qPrintable(tr("Hardware revision: %1\n").arg(service->hardwareRevision())), stdout);
118 fputs(qPrintable(tr("Firmware revision: %1\n").arg(service->firmwareRevision())), stdout);
119 fputs(qPrintable(tr("Software revision: %1\n").arg(service->softwareRevision())), stdout);
120 break;
121 }
122 QCoreApplication::quit(); // We're all done :)
123}
