80/*!
81 * Handles discovered Pokit devices, writing \a info to stdout.
82 */
83void ScanCommand::deviceDiscovered(const QBluetoothDeviceInfo &info)
84{
85 switch (format) {
86 case OutputFormat::Csv:
87 for (static bool firstTime = true; firstTime; firstTime = false) {
88 fputs(qPrintable(tr("uuid,address,name,major_class,minor_class,signal_strength\n")), stdout);
89 }
90 fputs(qPrintable(QString::fromLatin1("%1,%2,%3,%4,%5,%6\n").arg(info.deviceUuid().toString(),
91 info.address().toString(), escapeCsvField(info.name()), toString(info.majorDeviceClass()),
92 toString(info.majorDeviceClass(), info.minorDeviceClass())).arg(info.rssi())), stdout);
93 break;
94 case OutputFormat::Json:
95 fputs(QJsonDocument(toJson(info)).toJson(), stdout);
96 break;
97 case OutputFormat::Text:
98 fputs(qPrintable(tr("%1 %2 %3 %4\n").arg(info.deviceUuid().toString(),
99 info.address().toString(), info.name()).arg(info.rssi())), stdout);
100 break;
101 }
102}

126/*!
127 * Returns \a info as a JSON object.
128 */
129QJsonObject ScanCommand::toJson(const QBluetoothDeviceInfo &info) {
130 if (!info.isValid()) {
131 return QJsonObject();
132 }
133 QJsonObject json{
134 { QLatin1String("address"), info.address().toString() },
135 { QLatin1String("name"), info.name() },
136 { QLatin1String("isCached"), info.isCached() },
137 { QLatin1String("majorDeviceClass"), info.majorDeviceClass() },
138 { QLatin1String("majorDeviceClass"), toJson(info.majorDeviceClass()) },
139 { QLatin1String("minorDeviceClass"), toJson(info.majorDeviceClass(), info.minorDeviceClass()) },
140 { QLatin1String("signalStrength"), info.rssi() },
141 };
142 if (info.coreConfigurations() != QBluetoothDeviceInfo::UnknownCoreConfiguration) {
143 json.insert(QLatin1String("coreConfiguration"), toJson(info.coreConfigurations()));
144 }
145 if (!info.deviceUuid().isNull()) {
146 json.insert(QLatin1String("deviceUuid"), info.deviceUuid().toString());
147 }
148 #if (QT_VERSION >= QT_VERSION_CHECK(5, 12, 0)) // Added in Qt 5.12.
149 if (!info.manufacturerData().isEmpty()) {
150 json.insert(QLatin1String("manufacturerData"), toJson(info.manufacturerData()));
151 }
152 #endif
153 if (info.serviceClasses() != QBluetoothDeviceInfo::NoService) {
154 json.insert(QLatin1String("serviceClasses"), toJson(info.serviceClasses()));
155 }
156 if (!info.serviceUuids().isEmpty()) {
157 json.insert(QLatin1String("serviceUuids"), toJson(info.serviceUuids()));
158 }
159 return json;
160}

237/*!
238 * Returns Bluetooth manufacturer \a data as a JSON object that maps the manufacturer IDs (unsigned
239 * integers as strings) to arrays of one or more values.
240 */
241QJsonObject ScanCommand::toJson(const QMultiHash<quint16, QByteArray> &data)
242{
243 QJsonObject object;
244 QList<quint16> keys = data.uniqueKeys();
245 std::sort(keys.begin(), keys.end());
246 for (const quint16 key: keys) {
247 // Convert the key's values to a JSON array, reversing the order, because QMultiHash
248 // guarantees that the values are orderer "from the most recently inserted to the least
249 // recently inserted", which is the oppoosit of what we want.
250 QList<QByteArray> values = data.values(key);
251 std::reverse(values.begin(), values.end());
252 QJsonArray array;
253 for (const QByteArray &value: values) {
254 array.append(QLatin1String(value.toBase64()));
255 }
256 object.insert(QString::number(key), array);
257 }
258 return object;
259}
