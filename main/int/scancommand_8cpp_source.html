30/*!
31 * Construct a new ScanCommand object with \a parent.
32 */
33ScanCommand::ScanCommand(QObject * const parent) : AbstractCommand(parent)
34{
35 discoveryAgent = new PokitDeviceDiscoveryAgent(this);
36
37 connect(discoveryAgent, &PokitDeviceDiscoveryAgent::pokitDeviceDiscovered,
38 this, &ScanCommand::deviceDiscovered);
39
40 connect(discoveryAgent, &PokitDeviceDiscoveryAgent::finished, this, []() {
41 qCDebug(lc) << "Finished scanning for Pokit devices.";
42 QCoreApplication::quit();
43 });
44}

58QStringList ScanCommand::processOptions(const QCommandLineParser &parser)
59{
60 Q_ASSERT(discoveryAgent);
61
62 QStringList errors = AbstractCommand::processOptions(parser);
63 if (!errors.isEmpty()) {
64 return errors;
65 }
66
67 if (parser.isSet(QLatin1String("timeout"))) {
68 /// \todo Validate the value format.
69 discoveryAgent->setLowEnergyDiscoveryTimeout(
70 parser.value(QStringLiteral("timeout")).toInt()*1000);
71 qCDebug(lc) << tr("Set scan timeout to %1").arg(discoveryAgent->lowEnergyDiscoveryTimeout());
72 }
73
74 return errors;
75}

88/*!
89 * Handles discovered Pokit devices, writing \a info to stdout.
90 */
91void ScanCommand::deviceDiscovered(const QBluetoothDeviceInfo &info)
92{
93 switch (format) {
94 case OutputFormat::Csv:
95 fputs(qPrintable(tr("uuid,address,name,major_class,minor_class,signal_strength\n")), stdout);
96 fputs(qPrintable(tr("%1,%2,%3,%4,%5,%6\n").arg(info.deviceUuid().toString(),
97 info.address().toString(), escapeCsvField(info.name()), toString(info.majorDeviceClass()),
98 toString(info.majorDeviceClass(), info.minorDeviceClass())).arg(info.rssi())), stdout);
99 break;
100 case OutputFormat::Json:
101 fputs(QJsonDocument(toJson(info)).toJson(), stdout);
102 break;
103 case OutputFormat::Text:
104 fputs(qPrintable(tr("%1 %2 %3\n").arg(info.deviceUuid().toString(),
105 info.address().toString(), info.name())), stdout);
106 break;
107 }
108}
