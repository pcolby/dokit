38Informer::Informer(QObject * const parent) : QObject(parent), controller(nullptr)
39{
40 QBluetoothDeviceInfo info(QBluetoothAddress(QLatin1String("5C:02:72:09:AA:25")), QString(), 0);
41 qDebug() << toJsonObject(info);
42 controller = QLowEnergyController::createCentral(info);
43
44 connect(controller, &QLowEnergyController::serviceDiscovered, this, [this](const QBluetoothUuid &service) {
45 qDebug() << "service discovered" << service;
46 if (service == QBluetoothUuid(QStringLiteral(POKIT_SERVICE_STATUS))) {
47 qDebug() << "status";
48 QLowEnergyService * s = controller->createServiceObject(service);
49 qDebug() << s;
50 if (s != nullptr) {
51 s->discoverDetails();
52 }
53 }
54 });
55
56 QLowEnergyService * s = controller->createServiceObject(QBluetoothUuid(QLatin1String(POKIT_SERVICE_STATUS)), this);
57 qDebug() << "first service" << s;
58
59 connect(controller, &QLowEnergyController::connected, this, [this]() {
60 qDebug() << "connected" << controller->services().size();
61 controller->discoverServices();
62 });
63
64 connect(controller, &QLowEnergyController::disconnected, this, [this]() {
65 qDebug() << "disconnected" << controller->services().size();
66 });
67
68 connect(controller, &QLowEnergyController::discoveryFinished, this, [this]() {
69 qDebug() << "service discovery finished" << controller->services().size();
70 QLowEnergyService * s = controller->createServiceObject(QBluetoothUuid(QLatin1String(POKIT_SERVICE_STATUS)), this);
71 qDebug() << "second service" << s;
72 if (s) {
73 qDebug() << "second service" << s->serviceName() << s->serviceUuid();
74 #if (QT_VERSION < QT_VERSION_CHECK(6, 2, 0))
75 connect(s, QOverload<QLowEnergyService::ServiceError>::of(&QLowEnergyService::error),
76 #else
77 connect(s, &QLowEnergyService::errorOccurred,
78 #endif
79 this, [](const QLowEnergyService::ServiceError error) {
80 qCritical() << "ServcieError:" << error;
81 QCoreApplication::quit();
82 });
83
84 connect(s, &QLowEnergyService::stateChanged, this, [s](const QLowEnergyService::ServiceState state) {
85 qDebug() << "s1 state chnaged" << state;
86 #if (QT_VERSION < QT_VERSION_CHECK(6, 0, 0))
87 if (state == QLowEnergyService::ServiceDiscovered) {
88 #else
89 if (state == QLowEnergyService::RemoteServiceDiscovered) {
90 #endif
91 QLowEnergyCharacteristic c = s->characteristic(QUuid(POKIT_CHARACTERISTIC_STATUS_DEVICE));
92 qDebug() << c.name() << c.value().length() << c.value() << c.isValid();
93 if (c.isValid()) {
94 Q_ASSERT(c.value().length() == 20); /// \todo Proper error handling.
95
96 // Both of the following methods produce the same, correct results. Will
97 // settle on one when refactoring into library classes.
98
99 // The bytearray-indexing method.
100 qDebug() << "version major" << qFromLittleEndian<quint8>(c.value().mid(0,1));
101 qDebug() << "version minor" << qFromLittleEndian<quint8>(c.value().mid(1,1));
102 qDebug() << "max voltage" << qFromLittleEndian<quint16>(c.value().mid(2,2));
103 qDebug() << "max current" << qFromLittleEndian<quint16>(c.value().mid(4,2));
104 qDebug() << "max resistance" << qFromLittleEndian<quint16>(c.value().mid(6,2));
105 qDebug() << "max sample rate" << qFromLittleEndian<quint16>(c.value().mid(8,2));
106 qDebug() << "max buff size" << qFromLittleEndian<quint16>(c.value().mid(10,2));
107 qDebug() << "max cap mask" << qFromLittleEndian<quint16>(c.value().mid(12,2));
108 qDebug() << "mac addr" << c.value().mid(14);
109
110 // The datastream method.
111 QDataStream str(c.value());
112 str.setByteOrder(QDataStream::LittleEndian);
113 quint8 v1,v2;
114 quint16 maxV, maxC, maxR, maxS, buff, mask;
115 str >> v1 >> v2 >> maxV >> maxC >> maxR >> maxS >> buff >> mask;
116 qDebug() << "version major" << v1;
117 qDebug() << "version minor" << v2;
118 qDebug() << "max voltage" << maxV;
119 qDebug() << "max current" << maxC;
120 qDebug() << "max resistance" << maxR;
121 qDebug() << "max sample rate" << maxS;
122 qDebug() << "max buff size" << buff;
123 qDebug() << "max cap mask" << mask;
124 //qDebug() << "mac addr" << str;
125 }
126 }
127 });
128 s->discoverDetails();
129 }
130
131 QLowEnergyService * s2 = controller->createServiceObject(POKIT_SERVICE_DEVICE_INFO, this);
132 qDebug() << "second service s2" << s2;
133 if (s2) {
134 qDebug() << "second service s2" << s2->serviceName() << s2->serviceUuid();
135 connect(s2, &QLowEnergyService::stateChanged, this, [s2](const QLowEnergyService::ServiceState state) {
136 qDebug() << "s2 state chnaged" << state;
137 #if (QT_VERSION < QT_VERSION_CHECK(6, 0, 0))
138 if (state == QLowEnergyService::ServiceDiscovered) {
139 #else
140 if (state == QLowEnergyService::RemoteServiceDiscovered) {
141 #endif
142 QLowEnergyCharacteristic c = s2->characteristic(POKIT_CHARACTERISTIC_DEVICE_MANUFACTURER);
143 qDebug() << c.name() << c.value() << c.isValid();
144 c = s2->characteristic(POKIT_CHARACTERISTIC_DEVICE_MODEL_NUMBER);
145 qDebug() << c.name() << c.value() << c.isValid();
146 c = s2->characteristic(POKIT_CHARACTERISTIC_DEVICE_FIRMWARE_REV);
147 qDebug() << c.name() << c.value() << c.isValid();
148 c = s2->characteristic(POKIT_CHARACTERISTIC_DEVICE_SOFTWARE_REV);
149 qDebug() << c.name() << c.value() << c.isValid();
150 c = s2->characteristic(POKIT_CHARACTERISTIC_DEVICE_HARDWARE_REV);
151 qDebug() << c.name() << c.value() << c.isValid();
152 }
153 });
154 s2->discoverDetails();
155 }
156 });
157
158 #if (QT_VERSION < QT_VERSION_CHECK(6, 2, 0))
159 connect(controller, QOverload<QLowEnergyController::Error>::of(&QLowEnergyController::error),
160 #else
161 connect(controller, &QLowEnergyController::errorOccurred,
162 #endif
163 this, [](const QLowEnergyController::Error error) {
164 qCritical() << "Error:" << error;
165 QCoreApplication::quit();
166 });
167
168 qDebug() << "connecting";
169 controller->connectToDevice();
170}
