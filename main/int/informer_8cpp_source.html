35Informer::Informer(QObject * const parent) : QObject(parent)
36{
37 PokitDevice * device = new PokitDevice(QLatin1String("5C:02:72:09:AA:25"), this);
38 qDebug() << device;
39 StatusService * service = device->status();
40 qDebug() << service;
41
42 connect(device->controller(),
43 #if (QT_VERSION < QT_VERSION_CHECK(6, 2, 0))
44 QOverload<QLowEnergyController::Error>::of(&QLowEnergyController::error),
45 #else
46 &QLowEnergyController::errorOccurred,
47 #endif
48 [](QLowEnergyController::Error error) {
49 qDebug() << error;
50 ::exit(EXIT_FAILURE); ///< \todo Exit gracefully.
51 });
52
53 connect(service, &StatusService::serviceDetailsDiscovered,[service]() {

59 });
60
61 device->controller()->connectToDevice();
