55/*!
56 * Returns \c true if \a serviceUuids contains a known Pokit Status service UUID.
57 *
58 * Currently, this is the only known way to detect a Pokit device.
59 */
60bool isPokitProduct(const QList<QBluetoothUuid> &serviceUuids)
61{
62 return (serviceUuids.contains(StatusService::ServiceUuids::pokitMeter) ||
63 serviceUuids.contains(StatusService::ServiceUuids::pokitPro));
64}

67/*!
68 * Returns \c true if \a controller describes a Pokit device.
69 *
70 * Currently, this is based on whether or not \a controller's service UUIDs includes a known Pokit
71 * Status service, but this test criteria might be swapped for something else sometime.
72 *
73 * \see isPokitProduct

75bool isPokitProduct(const QLowEnergyController &controller)

77 return isPokitProduct(controller.services());

80/*!
81 * Returns the #PokitProduct corresponding to the Bluetooth \a serviceUuids.
82 *
83 * Currently, this is based on whether or not \a servceUuids includes a known Pokit
84 * Status service, but this test criteria might be swapped for something else sometime.
85 *
86 * \see isPokitProduct
87 */
88PokitProduct pokitProduct(const QList<QBluetoothUuid> &serviceUuids)
89{
90 if (serviceUuids.contains(StatusService::ServiceUuids::pokitMeter)) {
91 return PokitProduct::PokitMeter;
92 } else if (serviceUuids.contains(StatusService::ServiceUuids::pokitPro)) {
93 return PokitProduct::PokitPro;
94 } else {
95 Q_ASSERT_X(false, "pokitProduct", "device is not a Pokit product");
96 return PokitProduct::PokitMeter;
97 }
98}

105 *

108PokitProduct pokitProduct(const QLowEnergyController &controller)

115/// Encapsulates convenience functions for working with capacitance ranges.
116namespace CapacitanceRange {
117
118/*!
119 * Returns \a product's capacitance \a range as a human-friendly string.
120 *
121 * \note Since Pokit Meters do not support capacitance measurement, \a product should not be PokitProduct::PokitMeter.
122 *
123 * \see PokitPro::toString(const PokitPro::CapacitanceRange &range)
124 */
125QString toString(const PokitProduct product, const quint8 range)
126{
127 Q_ASSERT(product == PokitProduct::PokitPro);
128 switch (product) {
129 case PokitProduct::PokitMeter:
130 break;
131 case PokitProduct::PokitPro:
132 return PokitPro::toString(static_cast<PokitPro::CapacitanceRange>(range));
133 }
134 return QString();
135}
136
137/*!
138 * Returns the maximum value for \a product's \a range in (integer) nanofarads, or the string "Auto".
139 * If \a range is not a known valid value, then an null QVariant is returned.
140 *
141 * \note Since Pokit Meters do not support capacitance measurement, \a product should not be PokitProduct::PokitMeter.
142 *
143 * \see PokitPro::maxValue(const PokitPro::CapacitanceRange &range)
144 */
145QVariant maxValue(const PokitProduct product, const quint8 range)
146{
147 switch (product) {
148 case PokitProduct::PokitMeter:
149 break;
150 case PokitProduct::PokitPro:
151 return PokitPro::maxValue(static_cast<PokitPro::CapacitanceRange>(range));
152 }
153 return QString();
154}
