135AbstractPokitServicePrivate::AbstractPokitServicePrivate(const QBluetoothUuid &serviceUuid,
136 QLowEnergyController * controller, AbstractPokitService * const q)
137 : autoDiscover(true), controller(controller), service(nullptr), serviceUuid(serviceUuid),
138 q_ptr(q)
139{
140 if (controller) {
141 connect(controller, &QLowEnergyController::connected,
142 this, &AbstractPokitServicePrivate::connected);
143
144 connect(controller, &QLowEnergyController::discoveryFinished,
145 this, &AbstractPokitServicePrivate::discoveryFinished);
146
147 connect(controller, &QLowEnergyController::serviceDiscovered,
148 this, &AbstractPokitServicePrivate::serviceDiscovered);
149
150 createServiceObject();
151 }
152
153}

155/*!
156 * Creates an internal service object from the internal controller.
157 *
158 * Any existing service object will *not* be replaced.
159 *
160 * Returns \c true if a service was created successfully, either now, or sometime previously.
161 */
162bool AbstractPokitServicePrivate::createServiceObject()
163{
164 if (!controller) {
165 return false;
166 }
167
168 if (service) {
169 qCDebug(pokitService) << "Already have service object" << service;
170 return true;
171 }
172
173 service = controller->createServiceObject(serviceUuid, this);
174 if (!service) {
175 return false;
176 }
177 qCDebug(pokitService) << "Service object created" << service;
178
179 connect(service, &QLowEnergyService::stateChanged,
180 this, &AbstractPokitServicePrivate::stateChanged);
181 connect(service, &QLowEnergyService::characteristicRead,
182 this, &AbstractPokitServicePrivate::characteristicRead);
183 connect(service, &QLowEnergyService::characteristicWritten,
184 this, &AbstractPokitServicePrivate::characteristicWritten);
185
186 if (autoDiscover) {
187 service->discoverDetails();
188 }
189 return true;
190}

193/*!
194 * Read the \a uuid characteristic.
195 *
196 * If succesful, the `QLowEnergyService::characteristicRead` signal will be emitted by the internal
197 * service object. For convenience, derived classes should implement the characteristicRead()
198 * virtual function to handle the read value.
199 *
200 * Returns \c true if the characteristic read request was successfully queued, \c false otherwise.
201 *
202 * \see AbstractPokitService::readCharacteristics()
203 * \see AbstractPokitServicePrivate::characteristicRead()
204 */
205bool AbstractPokitServicePrivate::readCharacteristic(const QBluetoothUuid &uuid)
206{
207 if (!service) {
208 qCDebug(pokitService) << "Cannot read characteristic without a service object" << uuid;
209 return false;
210 }
211
212 const QLowEnergyCharacteristic characteristic = service->characteristic(uuid);
213 if (!characteristic.isValid()) {
214 qCDebug(pokitService) << "Cannot read characteristic that is not yet valid" << uuid;
215 return false;
216 }
217
218 qCDebug(pokitService) << "Reading characteristic" << uuid << characteristic.properties()
219 << characteristic.name() << characteristic.value();
220 service->readCharacteristic(characteristic);
221 return true;
222}

224/*!
225 * Handles `QLowEnergyController::connected` events.
226 *
227 * If `autoDiscover` is enabled, this will begin service discovery on the newly connected contoller.
228 *
229 * \see AbstractPokitService::autoDiscover()
230 */
231void AbstractPokitServicePrivate::connected()
232{
233 if (!controller) {
234 qCWarning(pokitService) << "Connected with no controller set" << sender();
235 return;
236 }
237
238 qCDebug(pokitService) << "Connected" << controller->remoteName()
239 << controller->remoteAddress() << controller->remoteDeviceUuid();
240 if (autoDiscover) {
241 controller->discoverServices();
242 }
243}

245/*!
246 * Handles `QLowEnergyController::discoveryFinished` events.
247 *
248 * As this event indicates that the conroller has finished discovering services, this function will
249 * invoke createServiceObject() to create the internal service object (if not already created).
250 */
251void AbstractPokitServicePrivate::discoveryFinished()
252{
253 if (!controller) {
254 qCWarning(pokitService) << "Discovery finished with no controller set" << sender();
255 return;
256 }
257
258 qCDebug(pokitService) << "Discovery finished" << controller->remoteName()
259 << controller->remoteAddress() << controller->remoteDeviceUuid();
260
261 if (!createServiceObject()) {
262 qCWarning(pokitService) << "Discovery finished, but service not found";
263 /// \todo Might need to emit an error signal here?
264 }
265}

267/*!
268 * Handles `QLowEnergyController::serviceDiscovered` events.
269 *
270 * If the discovered service is the one this (or rather the derived) class wraps, then
271 * createServiceObject() will be invoked immediately (otherwise it will be invoked after full
272 * service discovery has completed, ie in discoveryFinished()).
273 */
274void AbstractPokitServicePrivate::serviceDiscovered(const QBluetoothUuid &newService)
275{
276 if ((!service) && (newService == serviceUuid)) {
277 qCDebug(pokitService) << "Service discovered " << newService;
278 createServiceObject();
279 }
280}

282/*!
283 * Handles `QLowEnergyController::stateChanged` events.
284 *
285 * If \a newState indicates that service details have now been discovered, then
286 * AbstractPokitService::serviceDetailsDiscovered will be emitted.
287 *
288 * \see AbstractPokitService::autoDiscover()
289 */
290void AbstractPokitServicePrivate::stateChanged(QLowEnergyService::ServiceState newState)
291{
292 qCDebug(pokitService) << "State changed" << newState;
293
294 if (newState == QLowEnergyService::
295 #if (QT_VERSION < QT_VERSION_CHECK(6, 0, 0))
296 ServiceDiscovered
297 #else
298 RemoteServiceDiscovered
299 #endif
300 ) {
301 Q_Q(AbstractPokitService);
302 qCDebug(pokitService) << "Service details discovered";
303 emit q->serviceDetailsDiscovered();
304 }
305}
