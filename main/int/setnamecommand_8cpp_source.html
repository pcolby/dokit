51/*!
52 * \copybrief DeviceCommand::processOptions
53 *
54 * This implementation extends DeviceCommand::processOptions to process additional CLI options
55 * supported (or required) by this command.
56 */
57QStringList SetNameCommand::processOptions(const QCommandLineParser &parser)
58{
59 QStringList errors = DeviceCommand::processOptions(parser);
60 if (!errors.isEmpty()) {
61 return errors;
62 }
63
64 newName = parser.value(QLatin1String("new-name"));
65 if (newName.isEmpty()) {
66 errors.append(tr("New name cannot be empty."));
67 } else if (newName.length() > 11) {
68 errors.append(tr("New name cannot exceed 11 characters."));
69 }
70 return errors;
71}

73/*!
74 * Begins scanning for Pokit devices.
75 */
76bool SetNameCommand::start()
77{
78 Q_ASSERT(device);
79 if (!service) {
80 service = device->status();
81 Q_ASSERT(service);
82 connect(service, &StatusService::serviceDetailsDiscovered,
83 this, &SetNameCommand::serviceDetailsDiscovered);
84 connect(service, &StatusService::serviceErrorOccurred,
85 this, &SetNameCommand::serviceError);
86 connect(service, &StatusService::deivceNameWritten,
87 this, &SetNameCommand::deivceNameWritten);
88 }
89 qCInfo(lc).noquote() << tr("Connecting to device...");
90 device->controller()->connectToDevice();
91 return true;
92}

94/*!
95 * \copybrief DeviceCommand::serviceDetailsDiscovered
96 *
97 * This override sets the device's name, via the Pokit Status service.
98 */
99void SetNameCommand::serviceDetailsDiscovered()
100{
101 qCInfo(lc).noquote() << tr("Setting device name to: %1").arg(newName);
102 if (!service->setDeviceName(newName)) {
103 QCoreApplication::exit(EXIT_FAILURE);
104 }
105}

107/*!
108 * Handles StatusService::deivceNameWritten events, by outputting the result and exiting.
109 */
110void SetNameCommand::deivceNameWritten()
111{
112 switch (format) {
113 case OutputFormat::Csv:
114 fputs(qPrintable(tr("set_name_result\nsuccess\n")), stdout);
115 break;
116 case OutputFormat::Json:
117 fputs(qPrintable(QLatin1String("true\n")), stdout);
118 break;
119 case OutputFormat::Text:
120 fputs(qPrintable(tr("Done.\n")), stdout);
121 break;
122 }
123 QCoreApplication::quit();
124}
