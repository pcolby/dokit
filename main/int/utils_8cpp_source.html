34/*!
35 * Returns \a info as a JSON object.
36 */
37QJsonObject toJson(const QBluetoothDeviceInfo &info) {
38 QJsonObject json{
39 { QLatin1String("address"), info.address().toString() },
40 { QLatin1String("name"), info.name() },
41 { QLatin1String("isCached"), info.isCached() },
42 { QLatin1String("majorDeviceClass"), info.majorDeviceClass() },
43 { QLatin1String("majorDeviceClass"), toJson(info.majorDeviceClass()) },
44 { QLatin1String("minorDeviceClass"), toJson(info.majorDeviceClass(), info.minorDeviceClass()) },
45 { QLatin1String("signalStrength"), info.rssi() },
46 { QLatin1String("serviceUuids"), toJson(info.serviceUuids()) },
47
48 };
49 if (!info.deviceUuid().isNull()) {
50 json.insert(QLatin1String("deviceUuid"), info.deviceUuid().toString());
51 }
52 #if (QT_VERSION >= QT_VERSION_CHECK(5, 12, 0)) // Added in Qt 5.12.
53 if (!info.manufacturerData().isEmpty()) {
54 json.insert(QLatin1String("manufacturerData"), toJson(info.manufacturerData()));
55 }
56 #endif
57 if (info.serviceClasses() != QBluetoothDeviceInfo::NoService) {
58 json.insert(QLatin1String("serviceClasses"), toJson(info.serviceClasses()));
59 }
60 return json;
61}

63/*!
64 * Returns \a majorClass as a JSON value. This is equivalent to toString, except that if toString
65 * does not recognise \a majorClass, then \a majorClass is returned as a JSON number (not a string).
66 *
67 * \see toString(const QBluetoothDeviceInfo::MajorDeviceClass &majorClass)
68 */
69QJsonValue toJson(const QBluetoothDeviceInfo::MajorDeviceClass &majorClass)
70{
71 const QString string = toString(majorClass);
72 return (string.isNull() ? QJsonValue(majorClass) : QJsonValue(string));
73}
74
75/*!
76 * Returns \a minorClass as a JSON value. This is equivalent to toString, except that if toString
77 * does not recognise \a minorClass as a sub-class of \a majorClass, then \a minorClass is returned
78 * as a JSON number (not a string).
79 *
80 * \see toString(const QBluetoothDeviceInfo::MajorDeviceClass &majorClass, const quint8 minorClass)
81 */
82QJsonValue toJson(const QBluetoothDeviceInfo::MajorDeviceClass &majorClass, const quint8 minorClass)
83{
84 const QString string = toString(majorClass, minorClass);
85 return (string.isNull() ? QJsonValue(minorClass) : QJsonValue(string));
86}

109/*!

111 */
112QJsonArray toJson(const QList<QBluetoothUuid> &uuids)
113{
114 QJsonArray array;
115 for (const QBluetoothUuid &uuid: uuids) {
116 array.append(uuid.toString());

121/*!
122 * Returns Bluetooth manufacturer \a data as a JSON array.
123 */
124QJsonArray toJson(const QMultiHash<quint16, QByteArray> &data)
125{
126 QJsonArray array;
127 for (auto iter = data.cbegin(); iter != data.cend(); ++iter) {
128 array.append(QJsonObject{
129 { QLatin1String("manufacturerId"), iter.key() },
130 { QLatin1String("data"), QString::fromLatin1(iter.value().toBase64()) },
131 });
132 }
133 return array;
134}

136/*!
137 * Returns \a majorClass as a human-readable string, or a null QString if \a majorClass is not
138 * recognised.
139 *
140 * For example, if \a majorClass is \c QBluetoothDeviceInfo::ToyDevice, then the string `ToyDevice`
141 * is returned.
142 */
143QString toString(const QBluetoothDeviceInfo::MajorDeviceClass &majorClass)
144{
145 #define QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(value) \
146 if (majorClass == QBluetoothDeviceInfo::value) \
147 return QLatin1String(#value)
148 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(MiscellaneousDevice);
149 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(ComputerDevice);
150 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(PhoneDevice);
151 #if (QT_VERSION < QT_VERSION_CHECK(5, 13, 0))
152 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(LANAccessDevice); // Deprecated since Qt 5.13.
153 #else
154 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(NetworkDevice); // Added in Qt 5.13.
155 #endif
156 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(AudioVideoDevice);
157 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(PeripheralDevice);
158 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(ImagingDevice);
159 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(WearableDevice);
160 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(ToyDevice);
161 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(HealthDevice);
162 QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN(UncategorizedDevice);
163 #undef QTPOKIT_INTERNAL_IF_EQUAL_THEN_RETURN
164 qCDebug(pokitUtils) << "unknown major class" << majorClass;
165 return QString(); // Null QString indicates unknown minor class.
166}
