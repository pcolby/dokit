47void configureLogging(const QCommandLineParser &parser)
48{
49 // Start with the Qt default message pattern (see qtbase:::qlogging.cpp:defaultPattern)
50 QString messagePattern = QStringLiteral("%{if-category}%{category}: %{endif}%{message}");
51
52 if (parser.isSet(QStringLiteral("debug"))) {
53 #ifdef QT_MESSAGELOGCONTEXT
54 // %{file}, %{line} and %{function} are only available when QT_MESSAGELOGCONTEXT is set.
55 messagePattern.prepend(QStringLiteral("%{function} "));
56 #endif
57 messagePattern.prepend(QStringLiteral("%{time process} %{type} "));
58 QLoggingCategory::setFilterRules(QStringLiteral("pokit.*.debug=true"));
59 }
60
61 const QString color = parser.value(QStringLiteral("color"));
62 if ((color == QStringLiteral("yes")) || (color == QStringLiteral("auto") && haveConsole())) {
63 messagePattern.prepend(QStringLiteral(
64 "%{if-debug}\x1b[37m%{endif}" // White
65 "%{if-info}\x1b[32m%{endif}" // Green
66 "%{if-warning}\x1b[35m%{endif}" // Magenta
67 "%{if-critical}\x1b[31m%{endif}" // Red
68 "%{if-fatal}\x1b[31;1m%{endif}")); // Red and bold
69 messagePattern.append(QStringLiteral("\x1b[0m")); // Reset.
70 }
71
72 qSetMessagePattern(messagePattern);
73}

79void showCliError(const QString &errorText, const int exitCode = EXIT_FAILURE) {
80 // Output the same way QCommandLineParser does (qcommandlineparser.cpp::showParserMessage).
81 const QString message = QCoreApplication::applicationName() + QLatin1String(": ")
82 + errorText + QLatin1Char('\n');
83 fputs(qPrintable(message), stderr);
84 ::exit(exitCode);
85}

87Command getCliCommand(const QStringList &posArguments) {
88 if (posArguments.isEmpty()) {
89 return Command::None;
90 }
91 if (posArguments.size() > 1) {
92 showCliError(QObject::tr("More than one command: %1").arg(posArguments.join(QStringLiteral(", "))));
93 }
94
95 const QMap<QString, Command> supportedCommands {
96 { QStringLiteral("info"), Command::Info },
97 { QStringLiteral("status"), Command::Status },
98 { QStringLiteral("meter"), Command::Meter },
99 { QStringLiteral("dso"), Command::DSO },
100 { QStringLiteral("logger"), Command::Logger },
101 { QStringLiteral("scan"), Command::Scan },
102 { QStringLiteral("set-name"), Command::SetName },
103 { QStringLiteral("flash-led"), Command::FlashLed },
104 };
105 const Command command = supportedCommands.value(posArguments.first().toLower(), Command::None);
106 if (command == Command::None) {
107 showCliError(QObject::tr("Unknown command: %1").arg(posArguments.first()));
108 }
109 return command;
110}

112Command parseCommandLine(const QStringList &appArguments, QCommandLineParser &parser)
113{
114 // Setupt the command line options.
115 parser.addOptions({
116 {{QStringLiteral("debug")},
117 QCoreApplication::translate("parseCommandLine", "Enable debug output.")},
118 {{QStringLiteral("d"), QStringLiteral("device")},
119 QCoreApplication::translate("parseCommandLine", "Name or address of Pokit device(s) to use."),
120 QStringLiteral("name|addr")},
121 { QStringLiteral("color"),
122 QCoreApplication::translate("parseCommandLine", "Color the console output (default auto)."),
123 QStringLiteral("yes|no|auto"), QStringLiteral("auto")},
124 });
125 parser.addHelpOption();
126 parser.addOptions({
127 {{QStringLiteral("interval")},
128 QCoreApplication::translate("parseCommandLine","Update interval for meter and logger modes"),
129 QStringLiteral("n[m][s]")},
130 {{QStringLiteral("mode")},
131 QCoreApplication::translate("parseCommandLine","@todo Needs more work."),
132 QStringLiteral("AC|DC")},
133 {{QStringLiteral("new-name")},
134 QCoreApplication::translate("parseCommandLine","New name for Pokit device."),
135 QStringLiteral("name")},
136 {{QStringLiteral("output")},
137 QCoreApplication::translate("parseCommandLine","Format for output (default text)."),
138 QStringLiteral("text|json"), QStringLiteral("text")},
139 {{QStringLiteral("range")},
140 QCoreApplication::translate("parseCommandLine","Desired measurement range (default auto)."),
141 QStringLiteral("auto|n[m|K|M][V|A|O[hms]"), QStringLiteral("auto")},
142 {{QStringLiteral("samples")},
143 QCoreApplication::translate("parseCommandLine","Number of samples to acquire"),
144 QStringLiteral("samples")},
145 {{QStringLiteral("timeout")},
146 QCoreApplication::translate("parseCommandLine","@todo"),
147 QStringLiteral("seconds")},
148 {{QStringLiteral("trigger-level")},
149 QCoreApplication::translate("parseCommandLine","Level to trigger DSO acquisition."),
150 QStringLiteral("n[m][V|A]")},
151 {{QStringLiteral("trigger-mode")},
152 QCoreApplication::translate("parseCommandLine","Mode to trigger DSO acquisition (default free)."),
153 QStringLiteral("free|rising|falling"), QStringLiteral("free")},
154 });
155 parser.addVersionOption();
156 parser.addOptions({
157 {{QStringLiteral("window")},
158 QCoreApplication::translate("parseCommandLine","Sampling window for DSO acquisition"),
159 QStringLiteral("duration")},
160 });
161
162 // Add supported 'commands' (as positional arguments, so they'll appear in the help text).
163 parser.addPositionalArgument(QStringLiteral("info"),
164 QCoreApplication::translate("parseCommandLine", "Get Pokit device information"),
165 QStringLiteral(" "));
166 parser.addPositionalArgument(QStringLiteral("status"),
167 QCoreApplication::translate("parseCommandLine", "Get Pokit device status"),
168 QStringLiteral(" "));
169 parser.addPositionalArgument(QStringLiteral("meter"),
170 QCoreApplication::translate("parseCommandLine", "Access Pokit device's multimeter mode"),
171 QStringLiteral(" "));
172 parser.addPositionalArgument(QStringLiteral("dso"),
173 QCoreApplication::translate("parseCommandLine", "Access Pokit device's DSO mode"),
174 QStringLiteral(" "));
175 parser.addPositionalArgument(QStringLiteral("logger"),
176 QCoreApplication::translate("parseCommandLine", "Access Pokit device's data logger mode"),
177 QStringLiteral(" "));
178 parser.addPositionalArgument(QStringLiteral("scan"),
179 QCoreApplication::translate("parseCommandLine", "Scan Bluetooth for Pokit devices"),
180 QStringLiteral(" "));
181 parser.addPositionalArgument(QStringLiteral("set-name"),
182 QCoreApplication::translate("parseCommandLine", "Set Pokit device's name"),
183 QStringLiteral(" "));
184 parser.addPositionalArgument(QStringLiteral("flash-led"),
185 QCoreApplication::translate("parseCommandLine", "Flash Pokit device's LED"),
186 QStringLiteral(" "));
187
188 // Do the initial parse, the see if we have a command specified yet.
189 parser.parse(appArguments);
190 const Command command = getCliCommand(parser.positionalArguments());
191
192 // If we have a (single, valid) command, then remove the commands list from the help text.
193 if (command != Command::None) {
194 parser.clearPositionalArguments();
195 }
196
197 // Handle -h|--help explicitly, so we can tweak the output to include the <command> info.
198 if (parser.isSet(QStringLiteral("help"))) {
199 const QString commandString = (command == Command::None) ? QStringLiteral("<command>")
200 : parser.positionalArguments().constFirst();
201 fputs(qPrintable(parser.helpText()
202 .replace(QStringLiteral("[options]"), commandString + QStringLiteral(" [options]"))
203 .replace(QStringLiteral("Arguments:"), QStringLiteral("Command:"))
204 ), stdout);
205 ::exit(EXIT_SUCCESS);
206 }
207
208 // Process the command for real (ie throw errors for unknown options, etc).
209 parser.process(appArguments);
210 return command;
211}

213int main(int argc, char *argv[])
214{
215 QCoreApplication app(argc, argv);
216 app.setApplicationName(QStringLiteral(CMAKE_PROJECT_NAME));
217 app.setApplicationVersion(QStringLiteral(CMAKE_PROJECT_VERSION));
218
219 // Parse the command line.
220 const QStringList appArguments = app.arguments();
221 QCommandLineParser parser;
222 const Command command = parseCommandLine(appArguments, parser);
223 configureLogging(parser);
224
225 /// \todo Handle required and ignored options in a generalised fashion.
226
227 // Handle the given command.
228 switch (command) {
229 case Command::DSO: break;
230 case Command::FlashLed: break;
231 case Command::Info: {
232 const QStringList devices = parser.values(QLatin1String("device"));
233 for (const QString &device: devices) {
234 qDebug() << device;
235 Informer i(&app);
236 Q_UNUSED(i) // Just to make cppcheck happy for now.
237 return app.exec();
238 }
239 return EXIT_FAILURE;
240 } break;
241 case Command::Logger: break;
242 case Command::Meter: break;
243 case Command::None:
244 fputs("Missing argument: <command>\nSee --help for usage information.\n", stderr);
245 return EXIT_FAILURE;
246 case Command::Scan: {
247 Scanner scanner(&app);
248 scanner.start(parser.value(QStringLiteral("timeout")).toInt());
249 return app.exec();
250 }
251 case Command::Status: break;
252 case Command::SetName: break;
253 }
254 return app.exec();
255}
