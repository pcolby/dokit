52void configureLogging(const QCommandLineParser &parser)
53{
54 // Start with the Qt default message pattern (see qtbase:::qlogging.cpp:defaultPattern)
55 QString messagePattern = u"%{if-category}%{category}: %{endif}%{message}"_s;
56
57 if (parser.isSet(u"debug"_s)) {
58 #ifdef QT_MESSAGELOGCONTEXT
59 // %{file}, %{line} and %{function} are only available when QT_MESSAGELOGCONTEXT is set.
60 messagePattern.prepend(u"%{function} "_s);
61 #endif
62 messagePattern.prepend(u"%{time process} %{threadid} %{type} "_s);
63 QLoggingCategory::setFilterRules(u"dokit.*.debug=true\npokit.*.debug=true"_s);
64 }
65
66 if (const QString color = parser.value(u"color"_s);
67 (color == u"yes"_s) || (color == u"auto"_s && haveConsole())) {
68 messagePattern.prepend(QStringLiteral(
69 "%{if-debug}\x1b[37m%{endif}" // White
70 "%{if-info}\x1b[32m%{endif}" // Green
71 "%{if-warning}\x1b[35m%{endif}" // Magenta
72 "%{if-critical}\x1b[31m%{endif}" // Red
73 "%{if-fatal}\x1b[31;1m%{endif}")); // Red and bold
74 messagePattern.append(u"\x1b[0m"_s); // Reset.
75 }
76
77 qSetMessagePattern(messagePattern);
78}

96void showCliError(const QString &errorText)
97{
98 // Output the same way QCommandLineParser::showMessageAndExit() does.

100 fputs(qUtf8Printable(message), stderr);

103Command getCliCommand(const QStringList &posArguments)
104{
105 if (posArguments.isEmpty()) {
106 return Command::None;
107 }
108 if (posArguments.size() > 1) {
109 showCliError(Private::tr("More than one command: %1").arg(posArguments.join(u", "_s)));
110 ::exit(EXIT_FAILURE);
111 }
112
113 const QMap<QString, Command> supportedCommands {
114 { u"info"_s, Command::Info },
115 { u"status"_s, Command::Status },
116 { u"meter"_s, Command::Meter },
117 { u"dso"_s, Command::DSO },
118 { u"logger-start"_s, Command::LoggerStart },
119 { u"logger-stop"_s, Command::LoggerStop },
120 { u"logger-fetch"_s, Command::LoggerFetch },
121 { u"scan"_s, Command::Scan },
122 { u"set-name"_s, Command::SetName },
123 { u"set-torch"_s, Command::SetTorch },
124 { u"flash-led"_s, Command::FlashLed },
125 { u"calibrate"_s, Command::Calibrate },
126 };
127 const Command command = supportedCommands.value(posArguments.first().toLower(), Command::None);
128 if (command == Command::None) {
129 showCliError(Private::tr("Unknown command: %1").arg(posArguments.first()));
130 ::exit(EXIT_FAILURE);
131 }
132 return command;
133}

136{

144 {{u"d"_s, u"device"_s},
145 Private::tr("Set the name, hardware address or macOS UUID of Pokit device to use. If not specified, "

147 Private::tr("device")},
148 });

150 parser.addOptions({
151 {{u"interval"_s},
152 Private::tr("Set the update interval for DOS, meter and "
153 "logger modes. Suffixes such as 's' and 'ms' (for seconds and milliseconds) may be used. "
154 "If no suffix is present, the units will be inferred from the magnitide of the given "
155 "interval. If the option itself is not specified, a sensible default will be chosen "
156 "according to the selected command."),
157 Private::tr("interval")},
158 {{u"mode"_s},
159 Private::tr("Set the desired operation mode. For "
160 "meter, dso, and logger commands, the supported modes are: AC Voltage, DC Voltage, AC Current, "

164 "requires firmware v1.5 or later for Pokit devices to support it. For the set-torch command "

166 Private::tr("mode")},
167 {{u"new-name"_s},
168 Private::tr("Give the desired new name for the set-name command."), Private::tr("name")},
169 {{u"output"_s},
170 Private::tr("Set the format for output. Supported "
171 "formats are: CSV, JSON and Text. All are case insenstitve. The default is Text."),
172 Private::tr("format"),
173 Private::tr("text")},
174 {{u"range"_s},
175 Private::tr("Set the desired measurement range. Pokit "
176 "devices support specific ranges, such as 0 to 300mV. Specify the desired upper limit, "
177 "and the best range will be selected, or use 'auto' to enable the Pokit device's auto-"
178 "range feature. The default is 'auto'."),
179 Private::tr("range"), u"auto"_s},
180 {{u"samples"_s}, Private::tr("Set the number of samples to acquire."), Private::tr("count")},
181 {{u"temperature"_s},
182 Private::tr("Set the current ambient temperature for the calibration command."), Private::tr("degrees")},
183 {{u"timeout"_s},
184 Private::tr("Set the device discovery scan timeout. "
185 "Suffixes such as 's' and 'ms' (for seconds and milliseconds) may be used. "
186 "If no suffix is present, the units will be inferred from the magnitide of the given "
187 "interval. The default behaviour is no timeout."),
188 Private::tr("period")},
189 {{u"timestamp"_s},
190 Private::tr("Set the optional starting timestamp for data logging. Default to 'now'."),
191 Private::tr("period")},
192 {{u"trigger-level"_s}, Private::tr("Set the DSO trigger level."), Private::tr("level")},
193 {{u"trigger-mode"_s},
194 Private::tr("Set the DSO trigger mode. Supported modes are: free, rising and falling. The default is free."),
195 Private::tr("mode"), u"free"_s},
196 });
197 parser.addVersionOption();
198
199 // Add supported 'commands' (as positional arguments, so they'll appear in the help text).
200 parser.addPositionalArgument(u"info"_s, Private::tr("Get Pokit device information"), u" "_s);
201 parser.addPositionalArgument(u"status"_s, Private::tr("Get Pokit device status"), u" "_s);
202 parser.addPositionalArgument(u"meter"_s, Private::tr("Access Pokit device's multimeter mode"), u" "_s);
203 parser.addPositionalArgument(u"dso"_s, Private::tr("Access Pokit device's DSO mode"), u" "_s);
204 parser.addPositionalArgument(u"logger-start"_s, Private::tr("Start Pokit device's data logger mode"), u" "_s);
205 parser.addPositionalArgument(u"logger-stop"_s, Private::tr("Stop Pokit device's data logger mode"), u" "_s);
206 parser.addPositionalArgument(u"logger-fetch"_s, Private::tr("Fetch Pokit device's data logger samples"), u" "_s);
207 parser.addPositionalArgument(u"scan"_s, Private::tr("Scan Bluetooth for Pokit devices"), u" "_s);
208 parser.addPositionalArgument(u"set-name"_s, Private::tr("Set Pokit device's name"), u" "_s);
209 parser.addPositionalArgument(u"set-torch"_s, Private::tr("Set Pokit device's torch on or off"), u" "_s);
210 parser.addPositionalArgument(u"flash-led"_s, Private::tr("Flash Pokit device's LED (Pokit Meter only)"), u" "_s);
211 parser.addPositionalArgument(u"calibrate"_s, Private::tr("Calibrate Pokit device temperature"), u" "_s);
212
213 // Do the initial parse, the see if we have a command specified yet.
214 parser.parse(appArguments);
215 configureLogging(parser);
216 qCDebug(lc).noquote() << QCoreApplication::applicationName() << QCoreApplication::applicationVersion();
217 qCDebug(lc).noquote() << "Qt " QT_VERSION_STR " compile-time";
218 qCDebug(lc).noquote() << "Qt" << qVersion() << "runtime";
219 const Command command = getCliCommand(parser.positionalArguments());
220
221 // If we have a (single, valid) command, then remove the commands list from the help text.
222 if (command != Command::None) {
223 parser.clearPositionalArguments();
224 }
225
226 // Handle -h|--help explicitly, so we can tweak the output to include the <command> info.
227 if (parser.isSet(u"help"_s)) {
228 const QString commandString = (command == Command::None) ? u"<command>"_s
229 : parser.positionalArguments().constFirst();
230 std::cout << qUtf8Printable(parser.helpText()
231 .replace(u"[options]"_s, commandString + u" [options]"_s)
232 .replace(u"Arguments:"_s, u"Command:"_s)
233 );
234 ::exit(EXIT_SUCCESS);
235 }
236
237 // Process the command for real (ie throw errors for unknown options, etc).
238 parser.process(appArguments);
239 return command;
240}

242AbstractCommand * getCommandObject(const Command command, QObject * const parent)
243{
244 switch (command) {
245 case Command::None:
246 showCliError(Private::tr("Missing argument: <command>\nSee --help for usage information."));
247 return nullptr;
248 case Command::Calibrate: return new CalibrateCommand(parent);
249 case Command::DSO: return new DsoCommand(parent);
250 case Command::FlashLed: return new FlashLedCommand(parent);
251 case Command::Info: return new InfoCommand(parent);
252 case Command::LoggerStart: return new LoggerStartCommand(parent);
253 case Command::LoggerStop: return new LoggerStopCommand(parent);
254 case Command::LoggerFetch: return new LoggerFetchCommand(parent);
255 case Command::Meter: return new MeterCommand(parent);
256 case Command::Scan: return new ScanCommand(parent);
257 case Command::Status: return new StatusCommand(parent);
258 case Command::SetName: return new SetNameCommand(parent);
259 case Command::SetTorch: return new SetTorchCommand(parent);
260 }
261 showCliError(Private::tr("Unknown command (%1)").arg((int)command));
262 return nullptr;
263}

265int main(int argc, char *argv[])
266{
267 // Setup the core application.
268 QCoreApplication app(argc, argv);
269 QCoreApplication::setApplicationName(QStringLiteral(PROJECT_NAME));
270 QCoreApplication::setApplicationVersion(QString::fromLatin1(PROJECT_VERSION
271 #ifdef PROJECT_PRE_RELEASE
272 "-" PROJECT_PRE_RELEASE
273 #endif
274 #ifdef PROJECT_BUILD_ID
275 "+" PROJECT_BUILD_ID
276 #endif
277 ));
278
279#if defined(Q_OS_MACOS)
280 // Qt ignores shell locale overrides on macOS (QTBUG-51386), so mimic Qt's handling of LANG on *nixes.
281 const QString localeName = QString::fromLocal8Bit(qgetenv("LANG"));
282 if (!localeName.isEmpty()) {
283 const QLocale newLocale(localeName);
284 if (newLocale.name() != localeName.split('.'_L1).constFirst()) {
285 qWarning() << "Could not find locale" << localeName;
286 } else {
287 QLocale::setDefault(newLocale);
288 }
289 }
290#endif
291
292 // Install localised translators, if we have translations for the current locale.
293 const QLocale locale;
294 QTranslator appTranslator, libTranslator;
295 if (appTranslator.load(locale, u"cli"_s, u"/"_s, u":/i18n"_s)) {
296 QCoreApplication::installTranslator(&appTranslator);
297 }
298 if (libTranslator.load(locale, u"lib"_s, u"/"_s, u":/i18n"_s)) {
299 QCoreApplication::installTranslator(&libTranslator);
300 }
301
302 // Parse the command line.
303 const QStringList appArguments = QCoreApplication::arguments();
304 QCommandLineParser parser;
305 const Command commandType = parseCommandLine(appArguments, parser);
306 qCDebug(lc).noquote() << "Locale:" << locale << locale.uiLanguages();
307#if (QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)) // QTranslator::filePath() added in Qt 5.15.
308 qCDebug(lc).noquote() << "App translations:" <<
309 (appTranslator.filePath().isEmpty() ? u"<none>"_s : appTranslator.filePath());
310 qCDebug(lc).noquote() << "Library translations:" <<
311 (libTranslator.filePath().isEmpty() ? u"<none>"_s : libTranslator.filePath());
312#else
313 qCDebug(lc).noquote() << "App translations:" << (!appTranslator.isEmpty());
314 qCDebug(lc).noquote() << "Lib translations:" << (!libTranslator.isEmpty());
315#endif
316
317 // Handle the given command.
318 AbstractCommand * const command = getCommandObject(commandType, &app);
319 if (command == nullptr) {
320 return EXIT_FAILURE; // getCommandObject will have logged the reason already.
321 }
322 const QStringList cliErrors = command->processOptions(parser);
323 for (const QString &error: cliErrors) {
324 showCliError(error);
325 }
326 const int result = ((cliErrors.isEmpty()) && (command->start())) ? QCoreApplication::exec() : EXIT_FAILURE;
327 delete command; // We don't strictly need to do this, but it does fix QTBUG-119063, and is probably good practice.
328 return result;
329}
