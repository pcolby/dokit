53void configureLogging(const QCommandLineParser &parser)
54{
55 // Start with the Qt default message pattern (see qtbase:::qlogging.cpp:defaultPattern)
56 QString messagePattern = QStringLiteral("%{if-category}%{category}: %{endif}%{message}");
57
58 if (parser.isSet(QStringLiteral("debug"))) {
59 #ifdef QT_MESSAGELOGCONTEXT
60 // %{file}, %{line} and %{function} are only available when QT_MESSAGELOGCONTEXT is set.
61 messagePattern.prepend(QStringLiteral("%{function} "));
62 #endif
63 messagePattern.prepend(QStringLiteral("%{time process} %{type} "));
64 QLoggingCategory::setFilterRules(QStringLiteral("pokit.*.debug=true"));
65 }
66
67 const QString color = parser.value(QStringLiteral("color"));
68 if ((color == QStringLiteral("yes")) || (color == QStringLiteral("auto") && haveConsole())) {
69 messagePattern.prepend(QStringLiteral(
70 "%{if-debug}\x1b[37m%{endif}" // White
71 "%{if-info}\x1b[32m%{endif}" // Green
72 "%{if-warning}\x1b[35m%{endif}" // Magenta
73 "%{if-critical}\x1b[31m%{endif}" // Red
74 "%{if-fatal}\x1b[31;1m%{endif}")); // Red and bold
75 messagePattern.append(QStringLiteral("\x1b[0m")); // Reset.
76 }
77
78 qSetMessagePattern(messagePattern);
79}

96void showCliError(const QString &errorText) {
97 // Output the same way QCommandLineParser does (qcommandlineparser.cpp::showParserMessage).
98 const QString message = QCoreApplication::applicationName() + QLatin1String(": ")
99 + errorText + QLatin1Char('\n');
100 fputs(qPrintable(message), stderr);
101}

103Command getCliCommand(const QStringList &posArguments) {
104 if (posArguments.isEmpty()) {
105 return Command::None;
106 }
107 if (posArguments.size() > 1) {
108 showCliError(QObject::tr("More than one command: %1").arg(posArguments.join(QStringLiteral(", "))));
109 ::exit(EXIT_FAILURE);
110 }
111
112 const QMap<QString, Command> supportedCommands {
113 { QStringLiteral("info"), Command::Info },
114 { QStringLiteral("status"), Command::Status },
115 { QStringLiteral("meter"), Command::Meter },
116 { QStringLiteral("dso"), Command::DSO },
117 { QStringLiteral("logger-start"), Command::LoggerStart },
118 { QStringLiteral("logger-stop"), Command::LoggerStop },
119 { QStringLiteral("logger-fetch"), Command::LoggerFetch },
120 { QStringLiteral("scan"), Command::Scan },
121 { QStringLiteral("set-name"), Command::SetName },
122 { QStringLiteral("flash-led"), Command::FlashLed },
123 { QStringLiteral("calibrate"), Command::Calibrate },
124 };
125 const Command command = supportedCommands.value(posArguments.first().toLower(), Command::None);
126 if (command == Command::None) {
127 showCliError(QObject::tr("Unknown command: %1").arg(posArguments.first()));
128 ::exit(EXIT_FAILURE);
129 }
130 return command;
131}

229 QStringLiteral(" "));

231 QCoreApplication::translate("parseCommandLine", "Stop Pokit device's data logger mode"),
232 QStringLiteral(" "));
233 parser.addPositionalArgument(QStringLiteral("logger-fetch"),
234 QCoreApplication::translate("parseCommandLine", "Fetch Pokit device's data logger samples"),
235 QStringLiteral(" "));
236 parser.addPositionalArgument(QStringLiteral("scan"),
237 QCoreApplication::translate("parseCommandLine", "Scan Bluetooth for Pokit devices"),
238 QStringLiteral(" "));
239 parser.addPositionalArgument(QStringLiteral("set-name"),
240 QCoreApplication::translate("parseCommandLine", "Set Pokit device's name"),
241 QStringLiteral(" "));
242 parser.addPositionalArgument(QStringLiteral("flash-led"),
243 QCoreApplication::translate("parseCommandLine", "Flash Pokit device's LED"),
244 QStringLiteral(" "));
245 parser.addPositionalArgument(QStringLiteral("calibrate"),
246 QCoreApplication::translate("parseCommandLine", "Calibrate Pokit device temperature"),
247 QStringLiteral(" "));
248
249 // Do the initial parse, the see if we have a command specified yet.
250 parser.parse(appArguments);
251 configureLogging(parser);
252 const Command command = getCliCommand(parser.positionalArguments());
253
254 // If we have a (single, valid) command, then remove the commands list from the help text.
255 if (command != Command::None) {
256 parser.clearPositionalArguments();
257 }
258
259 // Handle -h|--help explicitly, so we can tweak the output to include the <command> info.
260 if (parser.isSet(QStringLiteral("help"))) {
261 const QString commandString = (command == Command::None) ? QStringLiteral("<command>")
262 : parser.positionalArguments().constFirst();
263 fputs(qPrintable(parser.helpText()
264 .replace(QStringLiteral("[options]"), commandString + QStringLiteral(" [options]"))
265 .replace(QStringLiteral("Arguments:"), QStringLiteral("Command:"))
266 ), stdout);
267 ::exit(EXIT_SUCCESS);
268 }
269
270 // Process the command for real (ie throw errors for unknown options, etc).
271 parser.process(appArguments);
272 return command;
273}

275AbstractCommand * getCommandObject(const Command command, QObject * const parent)
276{
277 switch (command) {
278 case Command::None:
279 showCliError(QCoreApplication::translate("main",
280 "Missing argument: <command>\nSee --help for usage information."));
281 return nullptr;
282 case Command::Calibrate: return new CalibrateCommand(parent);
283 case Command::DSO: return new DsoCommand(parent);
284 case Command::FlashLed: return new FlashLedCommand(parent);
285 case Command::Info: return new InfoCommand(parent);
286 case Command::LoggerStart: return new LoggerStartCommand(parent);
287 case Command::LoggerStop: return new LoggerStopCommand(parent);
288 case Command::LoggerFetch: return new LoggerFetchCommand(parent);
289 case Command::Meter: return new MeterCommand(parent);
290 case Command::Scan: return new ScanCommand(parent);
291 case Command::Status: return new StatusCommand(parent);
292 case Command::SetName: return new SetNameCommand(parent);
293 }
294 showCliError(QCoreApplication::translate("main", "Unknown command (%1)").arg((int)command));
295 return nullptr;
296}

298int main(int argc, char *argv[])
299{
300 // Setup the core application.
301 QCoreApplication app(argc, argv);
302 app.setApplicationName(QStringLiteral(CMAKE_PROJECT_NAME));
303 #ifdef POKIT_PRE_RELEASE
304 app.setApplicationVersion(QStringLiteral(CMAKE_PROJECT_VERSION "-" POKIT_PRE_RELEASE));
305 #else
306 app.setApplicationVersion(QStringLiteral(CMAKE_PROJECT_VERSION));
307 #endif
308
309 // Parse the command line.
310 const QStringList appArguments = app.arguments();
311 QCommandLineParser parser;
312 const Command commandType = parseCommandLine(appArguments, parser);
313
314 // Handle the given command.
315 AbstractCommand * const command = getCommandObject(commandType, &app);
316 if (command == nullptr) {
317 return EXIT_FAILURE; // getCommandObject will have logged the reason already.
318 }
319 const QStringList cliErrors = command->processOptions(parser);
320 for (const QString &error: cliErrors) {
321 showCliError(error);
322 }
323 if (!cliErrors.isEmpty()) {
324 return EXIT_FAILURE;
325 }
326 return (command->start()) ? app.exec() : EXIT_FAILURE;
327}
