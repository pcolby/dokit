72 * This implementation extends DeviceCommand::processOptions to process additional CLI options
73 * supported (or required) by this command.

75QStringList DsoCommand::processOptions(const QCommandLineParser &parser)
76{
77 QStringList errors = DeviceCommand::processOptions(parser);
78 if (!errors.isEmpty()) {
79 return errors;
80 }
81
82 // Parse the (required) mode option.
83 const QString mode = parser.value(QLatin1String("mode")).trimmed().toLower();
84 if (mode.startsWith(QLatin1String("ac v")) || mode.startsWith(QLatin1String("vac"))) {
85 settings.mode = DsoService::Mode::AcVoltage;
86 } else if (mode.startsWith(QLatin1String("dc v")) || mode.startsWith(QLatin1String("vdc"))) {
87 settings.mode = DsoService::Mode::DcVoltage;
88 } else if (mode.startsWith(QLatin1String("ac c")) || mode.startsWith(QLatin1String("aac"))) {
89 settings.mode = DsoService::Mode::AcCurrent;
90 } else if (mode.startsWith(QLatin1String("dc c")) || mode.startsWith(QLatin1String("aac"))) {
91 settings.mode = DsoService::Mode::DcCurrent;
92 } else {
93 errors.append(tr("Unknown DSO mode: %1").arg(parser.value(QLatin1String("mode"))));
94 return errors;
95 }
96
97 // Parse the (required) range option.
98 {
99 const QString value = parser.value(QLatin1String("range"));
100 QString unit; quint32 sensibleMinimum = 0;
101 switch (settings.mode) {
102 case DsoService::Mode::Idle:
103 Q_ASSERT(false); // Not possible, since the mode parsing above never allows Idle.
104 break;
105 case DsoService::Mode::DcVoltage:
106 case DsoService::Mode::AcVoltage:
107 unit = QLatin1String("V");
108 sensibleMinimum = 50; // mV.
109 break;
110 case DsoService::Mode::DcCurrent:
111 case DsoService::Mode::AcCurrent:
112 unit = QLatin1String("A");
113 sensibleMinimum = 5; // mA.
114 break;
115 }
116 Q_ASSERT(!unit.isEmpty());
117 const quint32 rangeMax = parseMilliValue(value, unit, sensibleMinimum);
118 if (rangeMax == 0) {
119 errors.append(tr("Invalid range value: %1").arg(value));
120 } else {
121 settings.range = lowestRange(settings.mode, rangeMax);
122 }
123 }

187/*!
188 * Returns the lowest \a mode range that can measure at least up to \a desired max, or AutoRange
189 * if no such range is available.
190 */
191DsoService::Range DsoCommand::lowestRange(
192 const DsoService::Mode mode, const quint32 desiredMax)
193{
194 DsoService::Range range{ DsoService::VoltageRange::_6V_to_12V };
195 switch (mode) {
196 case DsoService::Mode::Idle:
197 qCWarning(lc).noquote() << tr("Idle has no defined ranges.");
198 Q_ASSERT(false); // Should never have been called with this Idle mode.
199 break;
200 case DsoService::Mode::DcVoltage:
201 case DsoService::Mode::AcVoltage:
202 range.voltageRange = lowestVoltageRange(desiredMax);
203 break;
204 case DsoService::Mode::DcCurrent:
205 case DsoService::Mode::AcCurrent:
206 range.currentRange = lowestCurrentRange(desiredMax);
207 break;
208 default:
209 qCWarning(lc).noquote() << tr("No defined ranges for mode %1.").arg((quint8)mode);
210 Q_ASSERT(false); // Should never have been called with this invalid mode.
211 }
212 return range;
213}

220/*!
221 * Returns the lowest current range that can measure at least up to \a desired max, or AutoRange
222 * if no such range is available.
223 */
224DsoService::CurrentRange DsoCommand::lowestCurrentRange(const quint32 desiredMax)
225{
226 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_0_to_10mA)
227 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_10mA_to_30mA)
228 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_30mA_to_150mA)
229 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_150mA_to_300mA)
230 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_300mA_to_3A)
231 return DsoService::CurrentRange::_300mA_to_3A; // Out of range, so go with the biggest.
232}
