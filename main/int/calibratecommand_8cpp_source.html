55/*!
56 * \copybrief DeviceCommand::processOptions
57 *
58 * This implementation extends DeviceCommand::processOptions to process additional CLI options
59 * supported (or required) by this command.
60 */
61QStringList CalibrationCommand::processOptions(const QCommandLineParser &parser)
62{
63 QStringList errors = DeviceCommand::processOptions(parser);
64 if (!errors.isEmpty()) {
65 return errors;
66 }
67
68 const QString temperatureString = parser.value(QLatin1String("temperature"));
69 bool ok;
70 const float temperatureFloat = temperatureString.toFloat(&ok);
71 if (ok) {
72 temperature = temperatureFloat;
73 } else {
74 errors.append(tr("Unrecognised temperature format: %1").arg(temperatureString));
75 }
76 return errors;
77}

79/*!
80 * \copybrief DeviceCommand::getService
81 *
82 * This override returns a pointer to a CalibrationService object.
83 */
84AbstractPokitService * CalibrationCommand::getService()
85{
86 Q_ASSERT(device);
87 if (!service) {
88 service = device->calibration();
89 Q_ASSERT(service);
90 connect(service, &CalibrationService::temperatureCalibrated,
91 this, &CalibrationCommand::temperatureCalibrated);
92 }
93 return service;
94}

96/*!
97 * \copybrief DeviceCommand::serviceDetailsDiscovered
98 *
99 * This override sets the ambient temperature, via the Calibration service.
100 */
101void CalibrationCommand::serviceDetailsDiscovered()
102{
103 DeviceCommand::serviceDetailsDiscovered(); // Just logs consistently.
104 qCInfo(lc).noquote() << tr("Calibrating temperature at %1 degrees celcius...").arg(temperature);
105 if (!service->calibrateTemperature(0)) {
106 QCoreApplication::exit(EXIT_FAILURE);
107 }
108}

110/*!
111 * Handles CalibrationService::temperatureCalibrated events, by outputting the result and exiting.
112 */
113void CalibrationCommand::temperatureCalibrated()
114{
115 switch (format) {
116 case OutputFormat::Csv:
117 fputs(qPrintable(tr("calibration_result\nsuccess\n")), stdout);
118 break;
119 case OutputFormat::Json:
120 fputs(qPrintable(QLatin1String("true\n")), stdout);
121 break;
122 case OutputFormat::Text:
123 fputs(qPrintable(tr("Done.\n")), stdout);
124 break;
125 }
126 QCoreApplication::quit();
127}
