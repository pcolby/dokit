61/*!
62 * \copybrief DeviceCommand::processOptions
63 *
64 * This implementation extends DeviceCommand::processOptions to process additional CLI options
65 * supported (or required) by this command.
66 */
67QStringList LoggerCommand::processOptions(const QCommandLineParser &parser)
68{
69 QStringList errors = DeviceCommand::processOptions(parser);
70 if (!errors.isEmpty()) {
71 return errors;
72 }
73
74 // Parse the (required) mode option.
75 const QString mode = parser.value(QLatin1String("mode")).trimmed().toLower();
76 if (mode.startsWith(QLatin1String("ac v")) || mode.startsWith(QLatin1String("vac"))) {
77 settings.mode = DataLoggerService::Mode::AcVoltage;
78 } else if (mode.startsWith(QLatin1String("dc v")) || mode.startsWith(QLatin1String("vdc"))) {
79 settings.mode = DataLoggerService::Mode::DcVoltage;
80 } else if (mode.startsWith(QLatin1String("ac c")) || mode.startsWith(QLatin1String("aac"))) {
81 settings.mode = DataLoggerService::Mode::AcCurrent;
82 } else if (mode.startsWith(QLatin1String("dc c")) || mode.startsWith(QLatin1String("aac"))) {
83 settings.mode = DataLoggerService::Mode::DcCurrent;
84 } else {
85 errors.append(tr("Unknown logger mode: %1").arg(parser.value(QLatin1String("mode"))));
86 return errors;
87 }
88
89 // Parse the (required) range option.
90 {
91 const QString value = parser.value(QLatin1String("range"));
92 QString unit; quint32 sensibleMinimum = 0;
93 switch (settings.mode) {
94 case DataLoggerService::Mode::Idle:
95 Q_ASSERT(false); // Not possible, since the mode parsing above never allows Idle.
96 break;
97 case DataLoggerService::Mode::DcVoltage:
98 case DataLoggerService::Mode::AcVoltage:
99 unit = QLatin1String("V");
100 sensibleMinimum = 50; // mV.
101 break;
102 case DataLoggerService::Mode::DcCurrent:
103 case DataLoggerService::Mode::AcCurrent:
104 unit = QLatin1String("A");
105 sensibleMinimum = 5; // mA.
106 break;
107 }
108 Q_ASSERT(!unit.isEmpty());
109 const quint32 rangeMax = parseMilliValue(value, unit, sensibleMinimum);
110 if (rangeMax == 0) {
111 errors.append(tr("Invalid range value: %1").arg(value));
112 } else {
113 settings.range = lowestRange(settings.mode, rangeMax);
114 }
115 }
116
117 // Parse the interval option.
118 if (parser.isSet(QLatin1String("interval"))) {
119 const QString value = parser.value(QLatin1String("interval"));
120 const quint32 interval = parseWholeValue(value, QLatin1String("s"));
121 if (interval == 0) {
122 errors.append(tr("Invalid interval value: %1").arg(value));
123 } else {
124 settings.updateInterval = interval;
125 }
126 }
127
128 // Parse the samples option.
129 if (parser.isSet(QLatin1String("samples"))) {
130 const QString value = parser.value(QLatin1String("samples"));
131 QLocale locale; bool ok;
132 const int samples = locale.toInt(value, &ok);
133 if (!ok) {
134 errors.append(tr("Invalid number of samples: %1").arg(value));
135 } else {
136 numberOfSamplesToRead = samples;
137 }
138 }
139
140 /// \todo Parse the timestamp option.
141
142 return errors;
143}

150AbstractPokitService * LoggerCommand::getService()
151{
152 Q_ASSERT(device);

154 service = device->dataLogger();

162/*!
163 * \copybrief DeviceCommand::serviceDetailsDiscovered
164 *
165 * This override fetches the current device's status, and outputs it in the selected format.
166 */
167void LoggerCommand::serviceDetailsDiscovered()
168{
169 /// \todo Move this next block to a new DataLoggerService::toString(range, mode) function.
170 QString range;
171 switch (settings.mode) {
172 case DataLoggerService::Mode::Idle: break;
173 case DataLoggerService::Mode::DcVoltage:
174 case DataLoggerService::Mode::AcVoltage:
175 range = DataLoggerService::toString(settings.range.voltageRange);
176 break;
177 case DataLoggerService::Mode::DcCurrent:
178 case DataLoggerService::Mode::AcCurrent:
179 range = DataLoggerService::toString(settings.range.currentRange);
180 break;
181 }
182
183 DeviceCommand::serviceDetailsDiscovered(); // Just logs consistently.
184 qCInfo(lc).noquote() << tr("Logging %1, with range %2, every %L3ms.")
185 .arg(DataLoggerService::toString(settings.mode), range).arg(settings.updateInterval);
186 service->setSettings(settings);
187}

189/*!
190 * Returns the lowest \a mode range that can measure at least up to \a desired max, or AutoRange
191 * if no such range is available.
192 */
193DataLoggerService::Range LoggerCommand::lowestRange(
194 const DataLoggerService::Mode mode, const quint32 desiredMax)
195{
196 DataLoggerService::Range range{ DataLoggerService::VoltageRange::_6V_to_12V };
197 switch (mode) {
198 case DataLoggerService::Mode::Idle:
199 qCWarning(lc).noquote() << tr("Idle has no defined ranges.");
200 Q_ASSERT(false); // Should never have been called with this Idle mode.
201 break;
202 case DataLoggerService::Mode::DcVoltage:
203 case DataLoggerService::Mode::AcVoltage:
204 range.voltageRange = lowestVoltageRange(desiredMax);
205 break;
206 case DataLoggerService::Mode::DcCurrent:
207 case DataLoggerService::Mode::AcCurrent:
208 range.currentRange = lowestCurrentRange(desiredMax);
209 break;
210 default:
211 qCWarning(lc).noquote() << tr("No defined ranges for mode %1.").arg((quint8)mode);
212 Q_ASSERT(false); // Should never have been called with this invalid mode.
213 }
214 return range;
215}

222/*!
223 * Returns the lowest current range that can measure at least up to \a desired max, or AutoRange
224 * if no such range is available.
225 */
226DataLoggerService::CurrentRange LoggerCommand::lowestCurrentRange(const quint32 desiredMax)
227{
228 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_0_to_10mA)
229 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_10mA_to_30mA)
230 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_30mA_to_150mA)
231 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_150mA_to_300mA)
232 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, CurrentRange::_300mA_to_3A)
233 return DataLoggerService::CurrentRange::_300mA_to_3A; // Out of range, so go with the biggest.
234}

236/*!
237 * Returns the lowest voltage range that can measure at least up to \a desired max, or AutoRange
238 * if no such range is available.
239 */
240DataLoggerService::VoltageRange LoggerCommand::lowestVoltageRange(const quint32 desiredMax)

242 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, VoltageRange::_0_to_300mV)
243 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, VoltageRange::_300mV_to_2V)
244 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, VoltageRange::_2V_to_6V)
245 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, VoltageRange::_6V_to_12V)
246 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, VoltageRange::_12V_to_30V)
247 POKIT_APP_IF_LESS_THAN_RETURN(desiredMax, VoltageRange::_30V_to_60V)
248 return DataLoggerService::VoltageRange::_30V_to_60V; // Out of range, so go with the biggest.
249}

256void LoggerCommand::settingsWritten()
257{
258 qCDebug(lc).noquote() << tr("Settings written; starting data logger...");
259 connect(service, &DataLoggerService::metadataRead, this, &LoggerCommand::metadataRead);
260 connect(service, &DataLoggerService::samplesRead, this, &LoggerCommand::outputSamples);
261 service->beginMetadata();
262 service->beginSampling();
