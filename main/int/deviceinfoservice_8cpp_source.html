93bool DeviceInfoService::readCharacteristics()
94{
95 const bool r1 = readFirmwareRevisionCharacteristic();
96 const bool r2 = readHardwareRevisionCharacteristic();
97 const bool r3 = readSoftwareRevisionCharacteristic();
98 const bool r4 = readManufacturerCharacteristics();
99 const bool r5 = readModelNumberCharacteristic();
100 return (r1 && r2 && r3 && r4 && r5);
101}

178/*!
179 * Returns the most recent value of the `Device Info` service's `Manufacturer Name` characteristic.
180 *
181 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
182 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), then a
183 * null QString is returned.
184 */
185QString DeviceInfoService::manufacturer() const
186{
187 Q_D(const DeviceInfoService);
188 if (!d->service) {
189 qCDebug(pokitService).noquote() << tr("No device characteristics without a service object.");
190 return QString();
191 }
192
193 const QLowEnergyCharacteristic characteristic =
194 d->service->characteristic(CharacteristicUuids::manufacturerName);
195 if (!characteristic.isValid()) {
196 qCDebug(pokitService).noquote() << tr("Manufacturer Name characteristic not valid yet.");
197 return QString();
198 }
199
200 return QString::fromUtf8(characteristic.value());
201}

366/*!
367 * Implements AbstractPokitServicePrivate::characteristicRead to parse \a value, then emit a
368 * specialised signal, for each supported \a characteristic.
369 */
370void DeviceInfoServicePrivate::characteristicRead(const QLowEnergyCharacteristic &characteristic,
371 const QByteArray &value)
372{
373 qCDebug(pokitService).noquote() << tr("Read characteristic \"%1\" (%2) of size %3:")
374 .arg(characteristic.name(), characteristic.uuid().toString()).arg(value.size()) << value;
375 Q_Q(DeviceInfoService);
376
377 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::manufacturerName) {
378 const QString name = QString::fromUtf8(value);
379 qCDebug(pokitService).noquote() << tr("Manufacturer name: \"%1\"").arg(name);
380 emit q->manufacturerRead(name);
381 return;
382 }
383
384 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::modelNumber) {
385 const QString model = QString::fromUtf8(value);
386 qCDebug(pokitService).noquote() << tr("Model number: \"%1\"").arg(model);
387 emit q->modelNumberRead(model);
388 return;
389 }
390
391 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::hardwareRevision) {
392 const QString revision = QString::fromUtf8(value);
393 qCDebug(pokitService).noquote() << tr("Hardware revision: \"%1\"").arg(revision);
394 emit q->hardwareRevisionRead(revision);
395 return;
396 }
397
398 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::firmwareRevision) {
399 const QString revision = QString::fromUtf8(value);
400 qCDebug(pokitService).noquote() << tr("Firmware revision: \"%1\"").arg(revision);
401 emit q->firmwareRevisionRead(revision);
402 return;
403 }
404
405 if (characteristic.uuid() == DeviceInfoService::CharacteristicUuids::softwareRevision) {
406 const QString revision = QString::fromUtf8(value);
407 qCDebug(pokitService).noquote() << tr("Software revision: \"%1\"").arg(revision);
408 emit q->softwareRevisionRead(revision);
409 return;
410 }
411
412 qCWarning(pokitService).noquote() << tr("Unknown characteristic read for Device Info service")
413 << serviceUuid << characteristic.name() << characteristic.uuid();
414}
