66/*!
67 * Begins scanning for Pokit devices.
68 */
69bool FlashLedCommand::start()
70{
71 Q_ASSERT(device);
72 if (!service) {
73 service = device->status();
74 Q_ASSERT(service);
75 connect(service, &StatusService::serviceDetailsDiscovered,
76 this, &FlashLedCommand::serviceDetailsDiscovered);
77 connect(service, &StatusService::serviceErrorOccurred,
78 this, &FlashLedCommand::serviceError);
79 connect(service, &StatusService::deviceLedFlashed,
80 this, &FlashLedCommand::deviceLedFlashed);
81 }
82 qCInfo(lc).noquote() << tr("Connecting to device...");
83 device->controller()->connectToDevice();
84 return true;
85}

100/*!
101 * Handles StatusService::deviceLedFlashed events, by outputting the result and exiting.
102 */
103void FlashLedCommand::deviceLedFlashed()
104{
105 switch (format) {
106 case OutputFormat::Csv:
107 fputs(qPrintable(tr("flash_led_result\nsuccess\n")), stdout);
108 break;
109 case OutputFormat::Json:
110 fputs(qPrintable(QLatin1String("true\n")), stdout);
111 break;
112 case OutputFormat::Text:
113 fputs(qPrintable(tr("Done.\n")), stdout);
114 break;
115 }
116 QCoreApplication::quit();
117}
