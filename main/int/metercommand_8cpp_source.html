48/*!
49 * \copybrief DeviceCommand::processOptions
50 *
51 * This implementation extends DeviceCommand::processOptions to process additional CLI options
52 * supported (or required) by this command.
53 */
54QStringList MeterCommand::processOptions(const QCommandLineParser &parser)
55{
56 QStringList errors = DeviceCommand::processOptions(parser);
57 if (!errors.isEmpty()) {
58 return errors;
59 }
60
61 // Parse the (required) mode option.
62 const QString mode = parser.value(QLatin1String("mode")).trimmed().toLower();
63 if (mode.startsWith(QLatin1String("ac v")) || mode.startsWith(QLatin1String("vac"))) {
64 settings.mode = MultimeterService::Mode::AcVoltage;
65 } else if (mode.startsWith(QLatin1String("dc v")) || mode.startsWith(QLatin1String("vdc"))) {
66 settings.mode = MultimeterService::Mode::DcVoltage;
67 } else if (mode.startsWith(QLatin1String("ac c")) || mode.startsWith(QLatin1String("aac"))) {
68 settings.mode = MultimeterService::Mode::AcCurrent;
69 } else if (mode.startsWith(QLatin1String("dc c")) || mode.startsWith(QLatin1String("adc"))) {
70 settings.mode = MultimeterService::Mode::DcCurrent;
71 } else if (mode.startsWith(QLatin1String("res"))) {
72 settings.mode = MultimeterService::Mode::Resistance;
73 } else if (mode.startsWith(QLatin1String("dio"))) {
74 settings.mode = MultimeterService::Mode::Diode;
75 } else if (mode.startsWith(QLatin1String("cont"))) {
76 settings.mode = MultimeterService::Mode::Continuity;
77 } else if (mode.startsWith(QLatin1String("temp"))) {
78 settings.mode = MultimeterService::Mode::Temperature;
79 } else {
80 errors.append(tr("Unknown meter mode: %1").arg(parser.value(QLatin1String("mode"))));
81 return errors;
82 }
83
84 // Parse the interval option.
85 if (parser.isSet(QLatin1String("interval"))) {
86 const QString value = parser.value(QLatin1String("interval"));
87 const quint32 interval = parseMilliValue(value, QLatin1String("s"), 500);
88 if (interval == 0) {
89 errors.append(tr("Invalid interval value: %1").arg(value));
90 } else {
91 settings.updateInterval = interval;
92 }
93 }
94
95 // Parse the range option.
96 if (parser.isSet(QLatin1String("range"))) {
97 const QString value = parser.value(QLatin1String("range"));
98 const bool isAuto = (value.trimmed().compare(QLatin1String("auto"), Qt::CaseInsensitive) == 0);
99 QString unit; quint32 sensibleMinimum = 0;
100 switch (settings.mode) {
101 case MultimeterService::Mode::DcVoltage:
102 case MultimeterService::Mode::AcVoltage:
103 if (isAuto) {
104 settings.range.voltageRange = MultimeterService::VoltageRange::AutoRange;
105 }
106 unit = QLatin1String("V");
107 sensibleMinimum = 50; // mV.
108 break;
109 case MultimeterService::Mode::DcCurrent:
110 case MultimeterService::Mode::AcCurrent:
111 if (isAuto) {
112 settings.range.currentRange = MultimeterService::CurrentRange::AutoRange;
113 }
114 unit = QLatin1String("A");
115 sensibleMinimum = 5; // mA.
116 break;
117 case MultimeterService::Mode::Resistance:
118 if (isAuto) {
119 settings.range.resistanceRange = MultimeterService::ResistanceRange::AutoRange;
120 }
121 unit = QLatin1String("ohms");
122 sensibleMinimum = 0; // Unused.
123 break;
124 default:
125 qCInfo(lc).noquote() << tr("Ignoring range value: %1").arg(value);
126 }
127 if ((!unit.isEmpty()) && (!isAuto)) { // isEmpty indicates a mode that has no range option.
128 const quint32 rangeMax = (sensibleMinimum == 0)
129 ? parseWholeValue(value, unit) : parseMilliValue(value, unit, sensibleMinimum);
130 if (rangeMax == 0) {
131 errors.append(tr("Invalid range value: %1").arg(value));
132 } else {
133 settings.range = lowestRange(settings.mode, rangeMax);
134 }
135 }
136 }
137
138 // Parse the samples option.
139 if (parser.isSet(QLatin1String("samples"))) {
140 const QString value = parser.value(QLatin1String("samples"));
141 const quint32 samples = parseWholeValue(value, QLatin1String("S"));
142 if (samples == 0) {
143 errors.append(tr("Invalid samples value: %1").arg(value));
144 } else {
145 samplesToGo = samples;
146 }
147 }
148 return errors;
149}

156AbstractPokitService * MeterCommand::getService()
157{
158 Q_ASSERT(device);
159 if (!service) {
160 service = device->multimeter();
161 Q_ASSERT(service);
162 connect(service, &MultimeterService::settingsWritten,
163 this, &MeterCommand::settingsWritten);
164 }
165 return service;
166}

173void MeterCommand::serviceDetailsDiscovered()
174{
175 DeviceCommand::serviceDetailsDiscovered(); // Just logs consistently.
176 const QString range = MultimeterService::toString(settings.range, settings.mode);
177 qCInfo(lc).noquote() << tr("Measuring %1, with range %2, every %L3ms.").arg(
178 MultimeterService::toString(settings.mode),
179 (range.isNull()) ? QString::fromLatin1("N/A") : range).arg(settings.updateInterval);
180 service->setSettings(settings);
181}

183/*!
184 * Returns the lowest \a mode range that can measure at least up to \a desired max, or AutoRange
185 * if no such range is available.
186 */
187MultimeterService::Range MeterCommand::lowestRange(
188 const MultimeterService::Mode mode, const quint32 desiredMax)
189{
190 MultimeterService::Range range;
191 switch (mode) {
192 case MultimeterService::Mode::DcVoltage:
193 case MultimeterService::Mode::AcVoltage:
194 range.voltageRange = lowestVoltageRange(desiredMax);
195 break;
196 case MultimeterService::Mode::DcCurrent:
197 case MultimeterService::Mode::AcCurrent:
198 range.currentRange = lowestCurrentRange(desiredMax);
199 break;
200 case MultimeterService::Mode::Resistance:
201 range.resistanceRange = lowestResistanceRange(desiredMax);
202 break;
203 default:
204 qCWarning(lc).noquote() << tr("Mode does not support range.");
205 range.voltageRange = MultimeterService::VoltageRange::AutoRange;
206 }
207 return range;
208}
209
210#define POKIT_APP_IF_LESS_THAN_RETURN(value, label) \
211if (value <= MultimeterService::maxValue(MultimeterService::label).toUInt()) { \
212 return MultimeterService::label; \
213}

274/*!
275 * Outputs meter \a reading in the selected ouput format.
276 */
277void MeterCommand::outputReading(const MultimeterService::Reading &reading)
278{
279 QString status;
280 if (reading.status == MultimeterService::MeterStatus::Error) {
281 status = QLatin1String("Error");
282 } else switch (reading.mode) {
283 case MultimeterService::Mode::Idle:
284 break;
285 case MultimeterService::Mode::DcVoltage:
286 case MultimeterService::Mode::AcVoltage:
287 case MultimeterService::Mode::DcCurrent:
288 case MultimeterService::Mode::AcCurrent:
289 case MultimeterService::Mode::Resistance:
290 status = (reading.status == MultimeterService::MeterStatus::AutoRangeOn)
291 ? tr("Auto Range On") : tr("Auto Range Off");
292 break;
293 case MultimeterService::Mode::Continuity:
294 status = (reading.status == MultimeterService::MeterStatus::Continuity)
295 ? tr("Continuity") : tr("No continuity");
296 break;
297 case MultimeterService::Mode::Temperature:
298 case MultimeterService::Mode::Diode:
299 status = tr("Ok");
300 break;
301 }
302
303 QString units;
304 switch (reading.mode) {
305 case MultimeterService::Mode::Idle: break;
306 case MultimeterService::Mode::DcVoltage: units = QLatin1String("Vdc"); break;
307 case MultimeterService::Mode::AcVoltage: units = QLatin1String("Vac"); break;
308 case MultimeterService::Mode::DcCurrent: units = QLatin1String("Adc"); break;
309 case MultimeterService::Mode::AcCurrent: units = QLatin1String("Aac"); break;
310 case MultimeterService::Mode::Resistance: units = QString::fromUtf8(""); break;
311 case MultimeterService::Mode::Diode: break;
312 case MultimeterService::Mode::Continuity: break;
313 case MultimeterService::Mode::Temperature: units = QString::fromUtf8("C"); break;
314 }
315
316 QString range;
317 QVariant rangeMin, rangeMax;
318 switch (reading.mode) {
319 case MultimeterService::Mode::Idle: break;
320 case MultimeterService::Mode::DcVoltage:
321 case MultimeterService::Mode::AcVoltage:
322 range = MultimeterService::toString(reading.range.voltageRange);
323 rangeMin = MultimeterService::minValue(reading.range.voltageRange);
324 rangeMax = MultimeterService::maxValue(reading.range.voltageRange);
325 break;
326 case MultimeterService::Mode::DcCurrent:
327 case MultimeterService::Mode::AcCurrent:
328 range = MultimeterService::toString(reading.range.currentRange);
329 rangeMin = MultimeterService::minValue(reading.range.currentRange);
330 rangeMax = MultimeterService::maxValue(reading.range.currentRange);
331 break;
332 case MultimeterService::Mode::Resistance:
333 range = MultimeterService::toString(reading.range.resistanceRange);
334 rangeMin = MultimeterService::minValue(reading.range.resistanceRange);
335 rangeMax = MultimeterService::maxValue(reading.range.resistanceRange);
336 break;
337 case MultimeterService::Mode::Diode: break;
338 case MultimeterService::Mode::Continuity: break;
339 case MultimeterService::Mode::Temperature: break;
340 }
341
342 switch (format) {
343 case OutputFormat::Csv:
344 for (; showCsvHeader; showCsvHeader = false) {
345 std::cout << qUtf8Printable(tr("mode,value,units,status,range_min_milli,range_max_milli\n"));
346 }
347 std::cout << qUtf8Printable(QString::fromLatin1("%1,%2,%3,%4,%5,%6\n")
348 .arg(escapeCsvField(MultimeterService::toString(reading.mode)))
349 .arg(reading.value, 0, 'f').arg(units, status, rangeMin.toString(), rangeMax.toString())
350 );
351 break;
352 case OutputFormat::Json: {
353 QJsonObject jsonObject{
354 { QLatin1String("status"), status },
355 { QLatin1String("value"), qIsInf(reading.value) ?
356 QJsonValue(tr("Infinity")) : QJsonValue(reading.value) },
357 { QLatin1String("mode"), MultimeterService::toString(reading.mode) },
358 };
359 if ((!rangeMin.isNull()) || (!rangeMax.isNull())) {
360 jsonObject.insert(QLatin1String("range"), QJsonObject{
361 { QLatin1String("min"),
362 #if (QT_VERSION >= QT_VERSION_CHECK(6, 0, 0))
363 (rangeMin.typeId() == QMetaType::Int)
364 #else
365 (rangeMin.type() == QVariant::Int)
366 #endif
367 ? QJsonValue(rangeMin.toInt()/1000.0) : rangeMin.toJsonValue() },
368 { QLatin1String("max"),
369 #if (QT_VERSION >= QT_VERSION_CHECK(6, 0, 0))
370 (rangeMax.typeId() == QMetaType::Int) ?
371 #else
372 (rangeMax.type() == QVariant::Int) ?
373 #endif
374 QJsonValue(rangeMax.toInt()/1000.0) : rangeMax.toJsonValue() },
375 });
376 }
377 std::cout << QJsonDocument(jsonObject).toJson().toStdString();
378 } break;
379 case OutputFormat::Text:
380 std::cout << qUtf8Printable(tr("Mode: %1 (0x%2)\n").arg(MultimeterService::toString(reading.mode))
381 .arg((quint8)reading.mode,2,16,QLatin1Char('0')));
382 std::cout << qUtf8Printable(tr("Value: %1 %2\n").arg(reading.value,0,'f').arg(units));
383 std::cout << qUtf8Printable(tr("Status: %1 (0x%2)\n").arg(status)
384 .arg((quint8)reading.status,2,16,QLatin1Char('0')));
385 std::cout << qUtf8Printable(tr("Range: %1 (0x%2)\n").arg(range)
386 .arg((quint8)reading.range.voltageRange,2,16,QLatin1Char('0')));
387 break;
388 }
389
390 if ((samplesToGo > 0) && (--samplesToGo == 0)) {
391 if (device) disconnect(); // Will exit the application once disconnected.
392 }
393}
