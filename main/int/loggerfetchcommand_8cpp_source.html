42/*!
43 * \copybrief DeviceCommand::getService
44 *
45 * This override returns a pointer to a DataLoggerService object.
46 */
47AbstractPokitService * LoggerFetchCommand::getService()
48{
49 Q_ASSERT(device);
50 if (!service) {
51 service = device->dataLogger();
52 Q_ASSERT(service);
53// connect(service, &DataLoggerService::settingsWritten,
54// this, &LoggerFetchCommand::settingsWritten);
55 connect(service, &DataLoggerService::metadataRead, this, &LoggerFetchCommand::metadataRead);
56 connect(service, &DataLoggerService::samplesRead, this, &LoggerFetchCommand::outputSamples);
57 }
58 return service;
59}
