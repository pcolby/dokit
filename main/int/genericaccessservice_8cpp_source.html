118/*!
119 * Returns the most recent value of the `Generic Access` services's `Appearance` characteristic.
120 *
121 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
122 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), `0xFFFF`
123 * is returned.
124 *
125 * \note Pokit's Bluetooth API suggests the device's `Appearance` will always be 0 aka "Unknown
126 * Appearance", so this is probably not a very useful characteristic if you already know you are
127 * dealing with a Pokit device.
128 */
129quint16 GenericAccessService::appearance() const
130{
131 Q_D(const GenericAccessService);
132 if (!d->service) {
133 qCDebug(pokitService).noquote() << tr("No device name without a service object.");
134 return std::numeric_limits<quint16>::max();
135 }
136
137 const QLowEnergyCharacteristic characteristic =
138 d->service->characteristic(CharacteristicUuids::appearance);
139 if (!characteristic.isValid()) {
140 qCDebug(pokitService).noquote() << tr("Appearance characteristic not valid yet.");
141 return std::numeric_limits<quint16>::max();
142 }
143
144 return d->parseAppearance(characteristic.value());
145}

147/*!
148 * Returns the most recent value of the `Generic Access` services's `Device Name` characteristic.
149 *
150 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
151 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), then a
152 * null QString is returned.
153 */
154QString GenericAccessService::deviceName() const
155{
156 Q_D(const GenericAccessService);

158 qCDebug(pokitService).noquote() << tr("No device name without a service object.");

162 const QLowEnergyCharacteristic characteristic =

172/*!
173 * Set's the Pokit device's name to \a name.
174 *
175 * Returns `true` if the write request was successfully queued, `false` otherwise.
176 *
177 * Emits deivceNameWritten() if/when the \a name has been set.
178 */
179bool GenericAccessService::setDeviceName(const QString &name)
180{
181 Q_D(const GenericAccessService);
182 if (!d->service) {
183 qCDebug(pokitService).noquote() << tr("Cannot set device name without a service object.");
184 return false;
185 }
186
187 const QLowEnergyCharacteristic characteristic =
188 d->service->characteristic(CharacteristicUuids::name);
189 if (!characteristic.isValid()) {
190 qCDebug(pokitService).noquote() << tr("Name characteristic not valid yet.");
191 return false;
192 }
193
194 const QByteArray value = name.toUtf8();
195 if (value.length() > 11) {
196 qCWarning(pokitService).noquote() << tr("Device name \"%1\" is too long (%2 > 11 bytes):")
197 .arg(name).arg(value.length()) << value;
198 return false;
199 }
200
201 d->service->writeCharacteristic(characteristic, value);
202 return (d->service->error() != QLowEnergyService::ServiceError::CharacteristicWriteError);
203}

249/*!
250 * Parses the `Appearance` \a value. Returns `0xFFFF` if not valid.
251 */
252quint16 GenericAccessServicePrivate::parseAppearance(const QByteArray &value)
253{
254 if (value.size() < 2) {
255 qCWarning(pokitService).noquote() << tr("Invalid appearance size %1 for value:")
256 .arg(value.size()) << value;
257 return std::numeric_limits<quint16>::max();
258 }
259 if (value.size() > 2) {
260 qCWarning(pokitService).noquote() << tr("Appearance has %1 extra bytes:")
261 .arg(value.size()-2) << value.mid(2);
262 }
263 const quint16 appearance = qFromLittleEndian<quint16>(value);
264 qCDebug(pokitService).noquote() << tr("Appearance: %1.").arg(appearance);
265 return appearance;
266}

268/*!
269 * Implements AbstractPokitServicePrivate::characteristicRead to parse \a value, then emit a
270 * specialised signal, for each supported \a characteristic.
271 */
272void GenericAccessServicePrivate::characteristicRead(const QLowEnergyCharacteristic &characteristic,
273 const QByteArray &value)
274{
275 qCDebug(pokitService).noquote() << tr("Read characteristic \"%1\" (%2) of size %3:")
276 .arg(characteristic.name(), characteristic.uuid().toString()).arg(value.size()) << value;
277
278 Q_Q(GenericAccessService);
279 if (characteristic.uuid() == GenericAccessService::CharacteristicUuids::appearance) {
280 emit q->appearanceRead(parseAppearance(value));
281 return;
282 }
283
284 if (characteristic.uuid() == GenericAccessService::CharacteristicUuids::name) {
285 const QString deviceName = QString::fromUtf8(value);
286 qCDebug(pokitService).noquote() << tr("Device name: \"%1\"").arg(deviceName);
287 emit q->deviceNameRead(deviceName);
288 return;
289 }
290
291 qCWarning(pokitService).noquote() << tr("Unknown characteristic read for Generic Access Service")
292 << serviceUuid << characteristic.name() << characteristic.uuid();
293}

295/*!
296 * Implements AbstractPokitServicePrivate::characteristicWritten to parse \a newValue, then emit a
297 * specialised signal, for each supported \a characteristic.
298 */
299void GenericAccessServicePrivate::characteristicWritten(const QLowEnergyCharacteristic &characteristic,
300 const QByteArray &newValue)
301{
302 qCDebug(pokitService).noquote() << tr("Characteristic \"%1\" (%2) written, with new value:")
303 .arg(characteristic.name(), characteristic.uuid().toString()) << newValue;
304
305 Q_Q(GenericAccessService);
306 if (characteristic.uuid() == GenericAccessService::CharacteristicUuids::appearance) {
307 qCWarning(pokitService).noquote() << tr("Appearance haracteristic is read-only, but somehow written")
308 << serviceUuid << characteristic.name() << characteristic.uuid();
309 return;
310 }
311
312 if (characteristic.uuid() == GenericAccessService::CharacteristicUuids::name) {
313 emit q->deivceNameWritten();
314 return;
315 }
316
317 qCWarning(pokitService).noquote() << tr("Unknown characteristic written for Generic Access service")
318 << serviceUuid << characteristic.name() << characteristic.uuid();
319}
