202/*!
203 * Configures the Pokit device's multimeter mode.
204 *
205 * Returns `true` if the write request was successfully queued, `false` otherwise.
206 *
207 * Emits settingsWritten() if/when the \a settings have been writtem successfully.
208 */
209bool MultimeterService::setSettings(const Settings &settings)
210{
211 Q_D(const MultimeterService);
212 const QLowEnergyCharacteristic characteristic =
213 d->getCharacteristic(CharacteristicUuids::settings);
214 if (!characteristic.isValid()) {
215 return false;
216 }
217
218 const QByteArray value = MultimeterServicePrivate::encodeSettings(settings);
219 if (value.isNull()) {
220 return false;
221 }
222
223 d->service->writeCharacteristic(characteristic, value);
224 return (d->service->error() != QLowEnergyService::ServiceError::CharacteristicWriteError);
225}

227/*!
228 * Returns the most recent value of the `Multimeter` service's `Reading` characteristic.
229 *
230 * The returned value, if any, is from the underlying Bluetooth stack's cache. If no such value is
231 * currently available (ie the serviceDetailsDiscovered signal has not been emitted yet), then the
232 * returned MultimeterService::Reading::value member will be a quiet NaN, which can be checked like:
233 *
234 * ```
235 * const MultimeterService::Reading reading = multimeterService->reading();
236 * if (qIsNaN(reading.value)) {
237 * // Handle failure.
238 * }
239 * ```
240 */
241MultimeterService::Reading MultimeterService::reading() const
242{
243 Q_D(const MultimeterService);
244 const QLowEnergyCharacteristic characteristic =
245 d->getCharacteristic(CharacteristicUuids::reading);
246 return (characteristic.isValid()) ? MultimeterServicePrivate::parseReading(characteristic.value())
247 : Reading{ MeterStatus::Error, std::numeric_limits<float>::quiet_NaN(), Mode::Idle, 0 };
248}

312/*!
313 * Returns \a settings in the format Pokit devices expect.
314 */
315QByteArray MultimeterServicePrivate::encodeSettings(const MultimeterService::Settings &settings)
316{
317 static_assert(sizeof(settings.mode) == 1, "Expected to be 1 byte.");
318 static_assert(sizeof(settings.range) == 1, "Expected to be 1 byte.");
319 static_assert(sizeof(settings.updateInterval) == 4, "Expected to be 4 bytes.");
320
321 QByteArray value;
322 QDataStream stream(&value, QIODevice::WriteOnly);
323 stream.setByteOrder(QDataStream::LittleEndian);
324 stream.setFloatingPointPrecision(QDataStream::SinglePrecision); // 32-bit floats, not 64-bit.
325 stream << (quint8)settings.mode << settings.range << settings.updateInterval;
326
327 Q_ASSERT(value.size() == 6);
328 return value;
329}

331/*!
332 * Parses the `Reading` \a value into a MultimeterService::Reading struct.
333 */
334MultimeterService::Reading MultimeterServicePrivate::parseReading(const QByteArray &value)
335{
336 MultimeterService::Reading reading{
337 MultimeterService::MeterStatus::Error,
338 std::numeric_limits<float>::quiet_NaN(),
339 MultimeterService::Mode::Idle, 0
340 };
341
342 if (!checkSize(QLatin1String("Reading"), value, 7, 7)) {
343 return reading;
344 }
345
346 reading.status = MultimeterService::MeterStatus(value.at(0));
347 reading.value = qFromLittleEndian<float>(value.mid(1,4));
348 reading.mode = static_cast<MultimeterService::Mode>(value.at(5));
349 reading.range = static_cast<quint8>(value.at(6));
350 return reading;
351}

353/*!
354 * Implements AbstractPokitServicePrivate::characteristicRead to parse \a value, then emit a
355 * specialised signal, for each supported \a characteristic.
356 */
357void MultimeterServicePrivate::characteristicRead(const QLowEnergyCharacteristic &characteristic,
358 const QByteArray &value)
359{
360 AbstractPokitServicePrivate::characteristicRead(characteristic, value);
361
362 Q_Q(MultimeterService);
363 if (characteristic.uuid() == MultimeterService::CharacteristicUuids::reading) {
364 emit q->readingRead(parseReading(value));
365 return;
366 }
367
368 if (characteristic.uuid() == MultimeterService::CharacteristicUuids::settings) {
369 qCWarning(lc).noquote() << tr("Settings characteristic is write-only, but somehow read")
370 << serviceUuid << characteristic.name() << characteristic.uuid();
371 return;
372 }
373
374 qCWarning(lc).noquote() << tr("Unknown characteristic read for Multimeter service")
375 << serviceUuid << characteristic.name() << characteristic.uuid();
376}
