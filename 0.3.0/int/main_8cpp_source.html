43void configureLogging(const QCommandLineParser &parser)
44{
45 // Start with the Qt default message pattern (see qtbase:::qlogging.cpp:defaultPattern)
46 QString messagePattern = QStringLiteral("%{if-category}%{category}: %{endif}%{message}");
47
48 if (parser.isSet(QStringLiteral("debug"))) {
49 #ifdef QT_MESSAGELOGCONTEXT
50 // %{file}, %{line} and %{function} are only available when QT_MESSAGELOGCONTEXT is set.
51 messagePattern.prepend(QStringLiteral("%{function} "));
52 #endif
53 messagePattern.prepend(QStringLiteral("%{time process} %{threadid} %{type} "));
54 QLoggingCategory::setFilterRules(QStringLiteral("dokit.*.debug=true\npokit.*.debug=true"));
55 }
56
57 const QString color = parser.value(QStringLiteral("color"));
58 if ((color == QStringLiteral("yes")) || (color == QStringLiteral("auto") && haveConsole())) {
59 messagePattern.prepend(QStringLiteral(
60 "%{if-debug}\x1b[37m%{endif}" // White
61 "%{if-info}\x1b[32m%{endif}" // Green
62 "%{if-warning}\x1b[35m%{endif}" // Magenta
63 "%{if-critical}\x1b[31m%{endif}" // Red
64 "%{if-fatal}\x1b[31;1m%{endif}")); // Red and bold
65 messagePattern.append(QStringLiteral("\x1b[0m")); // Reset.
66 }
67
68 qSetMessagePattern(messagePattern);
69}

94Command getCliCommand(const QStringList &posArguments)
95{
96 if (posArguments.isEmpty()) {
97 return Command::None;
98 }
99 if (posArguments.size() > 1) {
100 showCliError(QCoreApplication::translate("getCliCommand", "More than one command: %1")
101 .arg(posArguments.join(QStringLiteral(", "))));
102 ::exit(EXIT_FAILURE);
103 }
104
105 const QMap<QString, Command> supportedCommands {
106 { QStringLiteral("info"), Command::Info },
107 { QStringLiteral("status"), Command::Status },
108 { QStringLiteral("meter"), Command::Meter },
109 { QStringLiteral("dso"), Command::DSO },
110 { QStringLiteral("logger-start"), Command::LoggerStart },
111 { QStringLiteral("logger-stop"), Command::LoggerStop },
112 { QStringLiteral("logger-fetch"), Command::LoggerFetch },
113 { QStringLiteral("scan"), Command::Scan },
114 { QStringLiteral("set-name"), Command::SetName },
115 { QStringLiteral("flash-led"), Command::FlashLed },
116 { QStringLiteral("calibrate"), Command::Calibrate },
117 };
118 const Command command = supportedCommands.value(posArguments.first().toLower(), Command::None);
119 if (command == Command::None) {
120 showCliError(QCoreApplication::translate("getCliCommand", "Unknown command: %1").arg(posArguments.first()));
121 ::exit(EXIT_FAILURE);
122 }
123 return command;
124}

231 QCoreApplication::translate("parseCommandLine", "Calibrate Pokit device temperature"),
232 QStringLiteral(" "));
233
234 // Do the initial parse, the see if we have a command specified yet.
235 parser.parse(appArguments);
236 configureLogging(parser);
237 const Command command = getCliCommand(parser.positionalArguments());
238
239 // If we have a (single, valid) command, then remove the commands list from the help text.
240 if (command != Command::None) {
241 parser.clearPositionalArguments();
242 }
243
244 // Handle -h|--help explicitly, so we can tweak the output to include the <command> info.
245 if (parser.isSet(QStringLiteral("help"))) {
246 const QString commandString = (command == Command::None) ? QStringLiteral("<command>")
247 : parser.positionalArguments().constFirst();
248 std::cout << qUtf8Printable(parser.helpText()
249 .replace(QStringLiteral("[options]"), commandString + QStringLiteral(" [options]"))
250 .replace(QStringLiteral("Arguments:"), QStringLiteral("Command:"))
251 );
252 ::exit(EXIT_SUCCESS);
253 }
254
255 // Process the command for real (ie throw errors for unknown options, etc).
256 parser.process(appArguments);
257 return command;
258}

260AbstractCommand * getCommandObject(const Command command, QObject * const parent)
261{
262 switch (command) {
263 case Command::None:
264 showCliError(QCoreApplication::translate("main",
265 "Missing argument: <command>\nSee --help for usage information."));
266 return nullptr;
267 case Command::Calibrate: return new CalibrateCommand(parent);
268 case Command::DSO: return new DsoCommand(parent);
269 case Command::FlashLed: return new FlashLedCommand(parent);
270 case Command::Info: return new InfoCommand(parent);
271 case Command::LoggerStart: return new LoggerStartCommand(parent);
272 case Command::LoggerStop: return new LoggerStopCommand(parent);
273 case Command::LoggerFetch: return new LoggerFetchCommand(parent);
274 case Command::Meter: return new MeterCommand(parent);
275 case Command::Scan: return new ScanCommand(parent);
276 case Command::Status: return new StatusCommand(parent);
277 case Command::SetName: return new SetNameCommand(parent);
278 }
279 showCliError(QCoreApplication::translate("main", "Unknown command (%1)").arg((int)command));
280 return nullptr;
281}

283int main(int argc, char *argv[])
284{
285 // Setup the core application.
286 QCoreApplication app(argc, argv);
287 QCoreApplication::setApplicationName(QStringLiteral(PROJECT_NAME));
288 QCoreApplication::setApplicationVersion(QString::fromLatin1(PROJECT_VERSION
289 #ifdef PROJECT_PRE_RELEASE
290 "-" PROJECT_PRE_RELEASE
291 #endif
292 #ifdef PROJECT_BUILD_ID
293 "+" PROJECT_BUILD_ID
294 #endif
295 ));
296
297 // Install localised translators, if we have translations for the current locale.
298 QTranslator appTranslator, libTranslator;
299 if (appTranslator.load(QLocale(), QStringLiteral("cli"), QStringLiteral("/"), QStringLiteral(":/i18n"))) {
300 QCoreApplication::installTranslator(&appTranslator);
301 }
302 if (libTranslator.load(QLocale(), QStringLiteral("lib"), QStringLiteral("/"), QStringLiteral(":/i18n"))) {
303 QCoreApplication::installTranslator(&libTranslator);
304 }
305
306 // Parse the command line.
307 const QStringList appArguments = QCoreApplication::arguments();
308 QCommandLineParser parser;
309 const Command commandType = parseCommandLine(appArguments, parser);
310 qCDebug(lc).noquote() << QCoreApplication::applicationName() << QCoreApplication::applicationVersion();
311 qCDebug(lc).noquote() << "Qt" << qVersion() << "(runtime) [" QT_VERSION_STR " compile-time]";
312#if (QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)) // QTranslator::filePath() added in Qt 5.15.
313 qCDebug(lc).noquote() << "App translations:" <<
314 (appTranslator.filePath().isEmpty() ? QStringLiteral("<none>") : appTranslator.filePath());
315 qCDebug(lc).noquote() << "Library translations:" <<
316 (libTranslator.filePath().isEmpty() ? QStringLiteral("<none>") : libTranslator.filePath());
317#else
318 qCDebug(lc).noquote() << "App translations:" << (!appTranslator.isEmpty());
319 qCDebug(lc).noquote() << "Lib translations:" << (!libTranslator.isEmpty());
320#endif
321
322 // Handle the given command.
323 AbstractCommand * const command = getCommandObject(commandType, &app);
324 if (command == nullptr) {
325 return EXIT_FAILURE; // getCommandObject will have logged the reason already.
326 }
327 const QStringList cliErrors = command->processOptions(parser);
328 for (const QString &error: cliErrors) {
329 showCliError(error);
330 }
331 if (!cliErrors.isEmpty()) {
332 return EXIT_FAILURE;
333 }
334 return (command->start()) ? QCoreApplication::exec() : EXIT_FAILURE;
335}
